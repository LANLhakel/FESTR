32Table::Table(const std::string &hydro_path, const std::string &table_path,
33 const std::string &table_fname): d()
34{
35 d.clear();
36 std::ifstream matfile((hydro_path + "materials.txt").c_str());
37 if (!matfile.is_open())
38 {
39 std::cerr << "Error: file " << hydro_path << "materials.txt "
40 << "is not open in "
41 << "Table::Table(parametrized)" << std::endl;
42 exit(EXIT_FAILURE);
43 }
44 std::string sin;
45 std::set<std::string> used_mat;
46 while (true)
47 {
48 matfile >> sin;
49 used_mat.insert(sin);
50 if (matfile.eof()) break;
51 }
52 matfile.close();
53 matfile.clear();
54
55 std::string fname(table_path + table_fname);
56 std::ifstream infile(fname.c_str());
57 if (!infile.is_open())
58 {
59 std::cerr << "Error: file " << fname << " is not open in "
60 << "Table::Table(parametrized)" << std::endl;
61 exit(EXIT_FAILURE);
62 }
63 utils::find_line(infile, "Material Z Atomic weight File handle");
64 std::string fin;
65 int zin;
66 double ain;
67 while (true)
68 {
69 infile >> sin;
70 if (infile.eof()) break;
71 infile >> zin >> ain >> fin;
72 if (used_mat.find(sin) != used_mat.end())
73 d.insert(TableMap::value_type(sin, TableData(zin, ain, fin) ));
74 }
75 infile.close();
76 infile.clear();
77}

143 const std::vector<double> &a) const
144{
145 size_t n = a.size();
