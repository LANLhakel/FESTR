40Cell::Cell(std::ifstream &material): np_unit("none"),
41 my_id(0), telow(-1.0), tehigh(-1.0), trlow(-1.0), trhigh(-1.0),
42 nmat(0), mat(), fplow(), fphigh(), nte(0), ntr(0), nfp(), nall(),
43 gte(), gtr(), gnp(), gfp(), tegrid(), trgrid(), fpgrid()
44{
45 utils::find_word(material, "Zone");
46 material >> my_id;
47
48 utils::find_word(material, "te");
49 material >> telow >> tehigh >> nte >> gte;
50 if (nte < 2) nte = 1;
51 tegrid = utils::get_grid(telow, tehigh, nte, gte, false);
52 nte = tegrid.size();
53 nall.emplace_back(nte);
54
55 utils::find_word(material, "tr");
56 material >> trlow >> trhigh >> ntr >> gtr;
57 if (ntr < 2) ntr = 1;
58 trgrid = utils::get_grid(trlow, trhigh, ntr, gtr, false);
59 ntr = trgrid.size();
60 nall.emplace_back(ntr);
61
62 utils::find_word(material, "nmat");
63 material >> nmat;
64 utils::find_line(material, "material densities (particles/cm3)");
65 size_t nproduct = 1;
66 for (unsigned short int i = 0; i < nmat; ++i)
67 {
68 std::string s, gx;
69 double xlow, xhigh;
70 size_t nx;
71 material >> s >> xlow >> xhigh >> nx >> gx;
72 mat.emplace_back(s);
73 fplow.emplace_back(xlow);
74 fphigh.emplace_back(xhigh);
75 if (nx < 2) nx = 1;
76 gfp.emplace_back(gx);
77 std::vector<double> fploc(utils::get_grid(xlow, xhigh, nx, gx, true));
78 fpgrid.insert( FpopMap::value_type(s, fploc) );
79 nx = fploc.size();
80 nproduct *= nx;
81 nfp.emplace_back(std::move(nx));
82 }
83 nall.emplace_back(std::move(nproduct));
84}

152CellEOS Cell::get_eos(const size_t ite, const size_t itr, const size_t ifp)
153{
154 CellEOS rv;
155 rv.te = tegrid.at(ite);
156 rv.tr = trgrid.at(itr);
157 rv.nmat = nmat;
158 try
159 {
160 if (nmat > 0)
161 {
162 std::vector<size_t> indx = utils::one_to_many(nfp, ifp);
163 size_t j;
164 std::string m;
165 for (size_t i = 0; i < nmat; ++i)
166 {
167 m = mat.at(i);
168 j = indx.at(i);
169 rv.fp.emplace_back(fpgrid[m].at(j));
170 rv.mat.emplace_back(std::move(m));
171 }
172 }
173 }
174 catch (const std::out_of_range &oor)
175 {
176 if (nmat != 0)
177 {
178 std::cerr << "Error in Cell.get_eos(): "
179 << "nfp is empty, but nmat is not zero" << std::endl;
180 exit(EXIT_FAILURE);
181 }
182 }
183 return rv;
184}
