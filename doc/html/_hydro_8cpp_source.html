34Hydro::Hydro(const bool analysis_in, const std::string &hydro_path,
35 const std::string &table_path, const std::string &table_fname,
36 const std::string &symmetry_in,
37 const double tmin_in, const double tmax_in):
38 path(hydro_path), tbl(hydro_path, table_path, table_fname),
39 ntimes(0), nintervals(0), time(), ntd(0), tmin(tmin_in), tmax(tmax_in),
40 time_index(), analysis(analysis_in), nzones(0), cell(), ndim(),
41 symmetry(symmetry_in)
42{
43 std::string fname;
44
45 if (analysis) // read the (one-and-only) Mesh for analysis
46 {
47 fname = path + "mesh_0.txt";
48 std::ifstream geometry(fname.c_str());
49 if (!geometry.is_open())
50 {
51 std::cerr << "Error: file " << fname << " is not open in "
52 << "Hydro::Hydro(parametrized)" << std::endl;
53 exit(EXIT_FAILURE);
54 }
55 utils::find_word(geometry, "Number_of_zones");
56 geometry >> nzones;
57 geometry.close();
58 geometry.clear();
59 cell.reserve(nzones);
60
61 fname = path + "time_0.txt";
62 std::ifstream material(fname.c_str());
63 if (!material.is_open())
64 {
65 std::cerr << "Error: file " << fname << " is not open in "
66 << "Hydro::Hydro(parametrized)" << std::endl;
67 exit(EXIT_FAILURE);
68 }
69
70 if (symmetry == "spherical")
71 nintervals = 0; // initialize for addition
72 else // symmetry == "none"
73 nintervals = 1; // initialize for multiplication
74
75 for (size_t i = 0; i < nzones; ++i)
76 {
77 auto cellptr = std::make_shared<Cell>(material);
78 size_t nx = cellptr->get_ncases();
79 cell.emplace_back(std::move(cellptr));
80 if (symmetry == "spherical")
81 nintervals += nx;
82 else // symmetry == "none"
83 nintervals *= nx;
84 ndim.emplace_back(std::move(nx));
85 }
86 ntimes = nintervals + 1;
87 time.emplace_back(0.0);
88 time.emplace_back(1.0);
89 time_index.emplace_back(0);
90 time_index.emplace_back(1);
91 ntd = 1;
92
93 material.close();
94 material.clear();
95
96 }
97 else // !analysis; read time instants for postprocessing
98 {
99 fname = hydro_path + "times.txt";
100 std::ifstream infile(fname.c_str());
101 if (!infile.is_open())
102 {
103 std::cerr << "Error: file " << fname << " is not open in "
104 << "Hydro::Hydro(parametrized)" << std::endl;
105 exit(EXIT_FAILURE);
106 }
107 utils::find_line(infile, "# start");
108 infile >> ntimes;
109 ntd = utils::ndigits(ntimes);
110 time.reserve(ntimes);
111 for (size_t i = 0; i < ntimes; ++i)
112 {
113 size_t j;
114 double t;
115 infile >> j >> t;
116 if (i != j)
117 {
118 std::cerr << "Error: time index mismatch while reading file "
119 << "times.txt: " << i << " != " << j << " in "
120 << "Hydro::Hydro(parametrized)" << std::endl;
121 exit(EXIT_FAILURE);
122 }
123 time.emplace_back(t);
124 if (t >= tmin && t <= tmax)
125 {
126 time_index.emplace_back(j);
127 ++nintervals;
128 }
129 }
130 --nintervals;
131 infile.close();
132 infile.clear();
133 } // analysis
134}

208void Hydro::load_at(const size_t i, Grid &g, Mesh &m) const
209{
210 std::string tlabel;
211 if (analysis) // analysis of case #i
212 {
213 tlabel = "0";
214 if (g.size() == 0) g.load(path, tlabel);
215 if (m.size() == 0) m.load(path, tlabel);
216
217 if (symmetry == "spherical")
218 {
219 std::pair<size_t, size_t> pr = utils::one_to_two(ndim, i);
220 size_t izone = pr.first;
221 size_t icase = pr.second;
222 auto cptr = cell.at(izone);
223 CellEOS e = cptr->get_cell_eos(icase);
224 auto zptr = m.get_zone(izone);
225 zptr->set_te(e.te);
226 zptr->set_tr(e.tr);
227 zptr->set_nmat(e.nmat);
228 zptr->set_mat(e.mat);
229 zptr->set_fp(e.fp);
230 }
231 else // symmetry == "none"
232 {
233 std::vector<size_t> indx = utils::one_to_many(ndim, i);
234 // seed Mesh with CellEOS data
235 for (size_t izone = 0; izone < nzones; ++izone)
236 {
237 auto cptr = cell.at(izone);
238 CellEOS e = cptr->get_cell_eos(indx.at(izone));
239 auto zptr = m.get_zone(izone);
240 zptr->set_te(e.te);
241 zptr->set_tr(e.tr);
242 zptr->set_nmat(e.nmat);
243 zptr->set_mat(e.mat);
244 zptr->set_fp(e.fp);
245 }
246 }
247 }
248 else // postprocessing time interval [i, i+1]
249 {
250 tlabel = utils::int_to_string(i, '0', ntd);
251 g.load(path, tlabel);
252 m.load(path, tlabel);
253 }
254}
