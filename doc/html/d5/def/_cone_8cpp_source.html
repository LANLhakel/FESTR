201 Vector3d Cone::velocity(const Grid &g, const Vector3d &w) const
202 {
203  Vector3d wrz(w.get_rz());
204  const double wx = w.getx();
205  const double wy = w.gety();
206  const double phi = atan2(wy, wx);
207 
208  const double VEC_SMALL = Vector3d::get_small();

210  double d;
211  double s = 0.0;
212  for (size_t i = 0; i < 2; ++i)
213  {
214  d = (POINT(i) - wrz).norm();
215  if (d < VEC_SMALL)
216  {
217  u = VELOCITY(i);
218  s = 1.0;
219  break;
220  }
221  d = 1.0 / d;
222  s += d;
223  u += d * VELOCITY(i);
