122  const double wy = w.gety();

188  {

190  {
191  double BIG = -Vector3d::get_big();
192  rv.t = BIG;
193  rv.w = Vector3d(BIG, BIG, BIG);
194  rv.is_found = false;
195  }
196  else
197  {
198  rv.t = (za - pz) / uz;
199  rv.w = p + u * rv.t;
200  rv.is_found = utils::sign_eqt(rv.t, eqt) == 1 && contains(g, rv.w);
201  }
202  }
203  else // pmh_2015_0508
204  {
205  Vector3d bhead = POINT(1);
206  double ra = atail.getx();
207  double px = p.getx();
208  double py = p.gety();

210  double dz = bhead.gety() - za; // Eq.(2)
211  double ux = u.getx();
212  double uy = u.gety();
213  double rp2 = px*px + py*py; // Eq.(6)
214  double zd = pz - za; // Eq.(7)
215  double dz2 = dz * dz;
216  double ff = dz2 * (px*ux + py*uy); // Eq.(8)
217  double gg = dz * ra * dr; // Eq.(9)
218  double hh = gg + zd * dr * dr; // Eq.(11)
219  double uzdr = uz * dr;

227  { // Ray line (defined by the p and u vectors) intersects z-axis (pmh_2022_0427)
228  // but is not horizontal
229  double phi = atan2(uy, ux);
230  Vector3d up = u.get_rz();

232  if (fabs(atan2(py, px) - phi) > cnst::PI/2) pp.setx(-pp.getx());

234  Vector3d a2(-atail.getx(), atail.gety(), 0.0);
235  Vector3d b2(-bhead.getx(), bhead.gety(), 0.0);
236  std::pair<bool, Vector3d> hit2rz = lines_intersection(pp, up, a2, b2, eqt);
237 
238  // initially assuming there is no solution
239  double BIG = -Vector3d::get_big();
240  rv.t = BIG;
241  rv.w = Vector3d(BIG, BIG, BIG);
242  rv.is_found = false;
243 
244  // 1st candidate
245  Vector3d hit1(BIG, BIG, BIG);
246  double t1 = BIG;
247  if (hit1rz.first)
248  {
249  Vector3d h1 = hit1rz.second;
250  hit1.setx(h1.getx() * cos(phi));
251  hit1.sety(h1.getx() * sin(phi));
252  hit1.setz(h1.gety());
253  Vector3d d1 = hit1 - p;
254  t1 = d1.norm() / u.norm() * (d1.unit() * u.unit());
255  }
256 
257  // 2nd candidate
258  Vector3d hit2(BIG, BIG, BIG);
259  double t2 = BIG;
260  if (hit2rz.first)
261  {
262  Vector3d h2 = hit2rz.second;
263  hit2.setx(h2.getx() * cos(phi));
264  hit2.sety(h2.getx() * sin(phi));
265  hit2.setz(h2.gety());
266  Vector3d d2 = hit2 - p;
267  t2 = d2.norm() / u.norm() * (d2.unit() * u.unit());
268  }
269 
270  // ensure t2 <= t1;
271  if (t2 > t1)
272  {
273  std::swap(t1, t2);
274  std::swap(hit1, hit2);
275  std::swap(hit1rz, hit2rz);
276  }
277

279  if (utils::sign_eqt(t1, eqt) == 1)
280  {
281  rv.t = t1;
282  rv.w = hit1;
283  rv.is_found = true;
284  }
285  if (rv.fid != fid && utils::sign_eqt(t2, eqt) == 1)
286  {
287  rv.t = t2;
288  rv.w = hit2;
289  rv.is_found = true;
290  }
291 
292  }
293  else
294  { // general case
295  // Eq.(12)
296  double a = dz2 * uxy2 - uzdr * uzdr;
297  double b = 2.0 * (ff - uz * hh);
298  double c = dz2 * (rp2 - ra*ra) - zd * (gg + hh);
299  #include <choose_root.inc>
300  }
301  }
302  else
303  { // avoid numerical problems for vertical rays
304  if (fabs(dr) < Vector3d::get_small())
305  { // no solution if *this is a cylinder
306  double BIG = -Vector3d::get_big();
307  rv.t = BIG;
308  rv.w = Vector3d(BIG, BIG, BIG);
309  rv.is_found = false;
310  }
311  else
312  {
313  rv.w.setx(px);
314  rv.w.sety(py);
315  rv.w.setz(za + (sqrt(rp2) - ra) * dz / dr);
316  rv.t = (rv.w.getz() - p.getz()) / uz;
317  rv.is_found = utils::sign_eqt(rv.t, eqt) == 1 && contains(g, rv.w);
318  }
319  }
320

322  double phi = atan2(rv.w.gety(), rv.w.getx());
323  double b_rad = bhead.getx();
324  Vector3d head(b_rad*cos(phi), b_rad*sin(phi), bhead.gety());
325  double a_rad = atail.getx();
326  Vector3d tail(a_rad*cos(phi), a_rad*sin(phi), atail.gety());
327  Vector3d cone_dir = (tail - head).normalize();
328  Vector3d cross_product = (cone_dir % u) / u.norm();
329  if (fabs(cross_product.norm()) < 1.0e-16)
330  { // Ray runs along *this
331  double BIG = -Vector3d::get_big();
332  rv.t = BIG;
333  rv.w = Vector3d(BIG, BIG, BIG);
334  rv.is_found = false;
335  }

338  // floating-point numerics near *this Cone's vertex
339  if (!rv.is_found && fid == rv.fid &&
340  sqrt(rp2) < 2*MINIMUM_DISTANCE) // short hop
341  { // move the Ray across the z-axis (tolerating a small error)
342  rv.w = Vector3d(-px, -py, pz);
343  // but this solution is provisional (don't get stuck here)
344  rv.t = Vector3d::get_big() / 2.0;
345  rv.is_found = true;
346  }
347  }
348  }

351  { // see, if the solution needs to be moved away from edge
352  // pmh_2019_1031
353  Vector3d w = rv.w;
354  double phi = atan2(w.gety(), w.getx());
355  std::pair<Vector3d, Vector3d> endpoints = get_endpoints(g, phi);
356  Vector3d a = endpoints.first;
357  Vector3d b = endpoints.second;

359  rv.w = linear_Vector3d_fit(MINIMUM_DISTANCE, a, b);
360  if ((w-b).norm() < MINIMUM_DISTANCE)
361  rv.w = linear_Vector3d_fit(MINIMUM_DISTANCE, b, a);
362  }

369 Vector3d Cone::velocity(const Grid &g, const Vector3d &w) const
370 {
371  Vector3d wrz(w.get_rz());
372  const double wx = w.getx();
373  const double wy = w.gety();
374  const double phi = atan2(wy, wx);
375 
376  const double VEC_SMALL = Vector3d::get_small();
377  Vector3d u;
378  double d;
379  double s = 0.0;
380  for (size_t i = 0; i < 2; ++i)
381  {
382  d = (POINT(i) - wrz).norm();
383  if (d < VEC_SMALL)
384  {
385  u = VELOCITY(i);
386  s = 1.0;
387  break;
388  }
389  d = 1.0 / d;
390  s += d;
391  u += d * VELOCITY(i);
392  }
393  u /= s;
394  double ux = u.getx() * cos(phi);
395  double uy = u.getx() * sin(phi);
396  return Vector3d(ux, uy, u.gety());
397 }
