71FaceID Mesh::next_face(const Grid &g, const RetIntercept &h) const
72{
73 if (h.fid.my_id == -1) // exit Face is the Surface of bounding Sphere Zone
74 {
75 auto z = get_zone(0); // bounding Sphere Zone
76 auto f = z->get_face(1); // inner Surface of *z
77 return f->get_neighbor(h.fid.my_zone); // 1-to-1 pairing of Faces here
78 }
79 else // regular Zone
80 { // or exiting bounding Sphere Zone through its outer Face (Sphere)
81 auto z = get_zone(h.fid.my_zone); // this Zone
82 auto f = z->get_face(h.fid.my_id); // exit Face of *z
83 size_t nf = f->num_nbr();
84 for (size_t i = 0; i < nf; ++i)
85 {
86 FaceID fid = f->get_neighbor(i);
87 if (fid.my_id == -1) // the neighbor is the bounding Sphere Zone
88 {
89 auto nz = get_zone(0);
90 FacePtr fz = nz->get_face(1);
91 SurfacePtr fs = std::dynamic_pointer_cast<Surface>(fz);
92 auto fn = fs->get_face(fid.my_zone);
93 if (fn->contains(g, h.w)) return FaceID(0, 1);
94 }
95 else // the neighbor is a regular Zone
96 {
97 auto nz = get_zone(fid.my_zone);
98 auto fn = nz->get_face(fid.my_id);
99 if (fn->contains(g, h.w)) return fid;
100 }
101 }
102 return FaceID(); // bounding Sphere's outer Face has no neighbors
103 }
104}

108void Mesh::load(const std::string &path, const std::string &tlabel)
109{
110 clear();
111 std::string fname(path + "mesh_" + tlabel + ".txt");
112 std::ifstream geometry(fname.c_str());
113 if (!geometry.is_open())
114 {
115 std::cerr << "Error: file " << fname << " is not open in "
116 << "Mesh::load" << std::endl;
117 exit(EXIT_FAILURE);
118 }
119 utils::find_word(geometry, "Number_of_zones");
120 geometry >> nzones;
121 zone.reserve(nzones);
122
123 fname = path + "time_" + tlabel + ".txt";
124 std::ifstream material(fname.c_str());
125 if (!material.is_open())
126 {
127 std::cerr << "Error: file " << fname << " is not open in "
128 << "Mesh::load" << std::endl;
129 exit(EXIT_FAILURE);
130 }
131
132 for (size_t i = 0; i < nzones; ++i)
133 {
134 utils::find_word(geometry, "Zone");
135 utils::find_word(material, "Zone");
136 zone.emplace_back(std::make_shared<Zone>(geometry, material));
137 }
138
139 geometry.close();
140 geometry.clear();
141 material.close();
142 material.clear();
143}
