466 void Detector::do_Ray(Progress *parent,
467  const IntPair &patch, const IntPair &direction,
468  const Grid &g, const Mesh &m, const Database &d,
469  const Table &tbl,
470  const std::string &froot, const std::string &hroot,
471  const std::string &tname, Goal &gol)
472 {
473  const size_t &ix = patch.first;
474  const size_t &iy = patch.second;
475 
476  Vector3d r0; // where the Ray hits this patch on *this Detector
477  if (symmetry == "none")
478  r0 = ro + ux * ix + uy * iy;
479  else if (symmetry == "spherical")
480  r0 = rc + ux * ix;
481 
482  double theta = static_cast<double>(direction.first) * dtheta;
483  double phi = static_cast<double>(direction.second) * dphi;
484  Vector3d cvec(local_to_global(-cnst::CV * Vector3d(theta, phi)));
485 
486  const double EQT = 1.0e-15;
487  const Face *f = m.get_zone(Zone::BOUNDING_ZONE)->get_face(0);
488  FaceID fid(Zone::BOUNDING_ZONE, -3);
489  RetIntercept pt = f->intercept(g, r0, cvec, EQT, fid);
490 
491  // set up Progress counter
492  int next_level;
493  size_t next_freq;
494  if (parent == NULL)
495  {
496  next_level = 0;
497  next_freq = 0;
498  }
499  else
500  {
501  next_level = parent->get_level() + 1;
502  next_freq = parent->get_next_freq(freq_trace);
503  }
504 
505  std::string cname(froot + omega_fname(direction) + "_" + tname);
506  std::string fname(path + cname);
507  std::string header(cname + "\n" + hroot + omega_string(direction));
508  Ray ray(next_level, next_freq, nhv, jmin, jmax, tracking, r0, cvec,
509  gol.get_analysis(), fname, header);
510  fname += ".txt";
511 
512  if (pt.is_found)
513  {
514  ray.diag_id = my_id;
515  ray.patch_id = patch;
516  ray.bundle_id = direction;
517  ray.trace(g, m);
518  ray.set_backlighter(back_type, back_value, hv);
519  ray.cross_Mesh(m, d, tbl, symmetry, ix);
520  if (ntheta == 0) // parallel Rays; one Ray per patch (W/cm2/sr/eV)
521  yp[patch] = ray.y;
522  else // bundle of Rays per patch (W/cm2/eV)
523  {
524  double domega; // pmh_2016_1013
525  if (direction == INT_PAIR_00) // spherical cap
526  domega = cnst::FOUR_PI * pow(sin(dtheta/4.0), 2);
527  else // spherical rectangle
528  domega = 2.0 * dphi * sin(theta) * sin(dtheta2);
529  yp[patch] += ray.y * (domega * ez.cos_angle(ray.v));
530  }
531  }
532 
533  ArrDbl ybroad(nhv);
534  utils::convolution(fwhm, hv, ray.y, hv, ybroad);
535  if (gol.get_analysis())
536  {
537  header += "\ndata in W/cm2/sr/eV";
538  gol.fit_eval(cname, ybroad);
539  }
540  else
541  {
542  double scale = gol.get_best_scale(cname);
543  if (utils::sign_eqt(scale - 1.0, cnst::SCALE_EQT) == 0)
544  header += "\ndata in W/cm2/sr/eV";
545  else
546  {
547  header += "\ndata in arbitrary_units";
548  ybroad *= scale;
549  }
550  ybroad.to_file(fname, header);
551  }
552 }

556 void Detector::do_patch(Progress *parent, const IntPair &patch,
557  const Grid &g, const Mesh &m, const Database &d,
558  const Table &tbl,
559  const size_t it, const double t, const double dt,
560  const int ntd, const bool peel_onion, Goal &gol)
561 {
562  // set up Progress counter
563  int next_level;
564  size_t next_freq;
565  if (parent == NULL)
566  {
567  next_level = 0;
568  next_freq = 0;
569  }
570  else
571  {
572  next_level = parent->get_level() + 1;
573  next_freq = parent->get_next_freq(freq_Ray);
574  }
575  const std::string SEP("----------------------------------------\n");
576  size_t ncount(ntheta == 0 ? 1 : (ntheta-1)*nphi + 1);
577  Progress counter(dname + "_Ray", next_level, ncount, next_freq,
578  SEP, std::cout);
579 
580  yp[patch].fill(0.0);
581  std::string cname(dname + "-yp" + patch_fname(patch));
582  std::string fname(path + cname);
583  std::string tname(time_fname(it, ntd));
584  std::string header(time_string(it, ntd, t));
585  header += patch_string(patch);
586  do_Ray(&counter, patch, INT_PAIR_00, g, m, d, tbl,
587  cname, header, tname, gol);
588  counter.advance();
589  for (size_t itheta = 1; itheta < ntheta; ++itheta)
590  for (size_t iphi = 0; iphi < nphi; ++iphi)
591  {
592  do_Ray(&counter, patch, IntPair(itheta, iphi), g, m, d, tbl,
593  cname, header, tname, gol);
594  counter.advance();
595  }
596 
597  if (ntheta > 0) // (W/eV); (W/cm2/sr/eV) for ntheta == 0
598  {
599  if (symmetry == "none")
600  yp[patch] *= da;
601  else if (symmetry == "spherical") // Eq.(2)
602  {
603  double darea;
604  if (patch.first == 0) // ix == 0 (central circle)
605  darea = cnst::PI * pow(dx/2.0, 2);
606  else // ix > 0 (circular strip of width dx)
607  darea = cnst::TWO_PI * static_cast<double>(patch.first) * pow(dx, 2);
608  yp[patch] *= darea;
609  }
610  }
611 
612  #ifndef MPI
613  yt[patch] += yp[patch] * dt; // (J/eV); (J/cm2/sr/eV) for ntheta == 0
614  #endif
615 
616  header = cname + "_" + tname + "\n" + header;
617  fname += "_" + tname + ".txt";
618 
619  ArrDbl ybroad(nhv);
620  utils::convolution(fwhm, hv, yp[patch], hv, ybroad);
621  if (gol.get_analysis())
622  {
623  if (ntheta == 0)
624  header += "data in W/cm2/sr/eV";
625  else
626  header += "data in W/eV";
627 
628  if (peel_onion)
629  {
630  // one Ray per Zone for now
631  Objective *obj =
632  const_cast<Objective *>(gol.get_objective(patch.first));
633  obj->update_best(ybroad, nhv, it);
634  }
635  else
636  gol.fit_eval(cname, ybroad);
637  }
638  else // postprocessing
639  {
640  double scale = gol.get_best_scale(cname);
641  if (utils::sign_eqt(scale - 1.0, cnst::SCALE_EQT) == 0)
642  {
643  if (ntheta == 0)
644  header += "data in W/cm2/sr/eV";
645  else
646  header += "data in W/eV";
647  }
648  else
649  {
650  header += "data in arbitrary_units";
651  ybroad *= scale;
652  }
653  ybroad.to_file(fname, header);
654  }
655 }

666  ys.fill(0.0);
667  double xi(0.0);
668  for (size_t ix = 0; ix < nx; ++ix)
669  for (size_t iy = 0; iy < ny; ++iy)
670  {
671  IntPair patch(IntPair(ix, iy));
672  do_patch(&counter, patch, g, m, d, tbl, it, t, dt, ntd,
673  false, gol);
674  if (symmetry == "none")
675  ys += yp[patch]; // (W/eV); (W/cm2/sr/eV) for ntheta == 0
676  else if (symmetry == "spherical") // Eq.(2)
677  {
678  if (ntheta == 0)
679  {
680  if (ix == 0) // dx = diameter of the central circle
681  ys = yp[patch] / 8.0;
682  else // ring width = r>(ix) - r<(ix) = dx, for ix > 0
683  ys += yp[patch] * xi; // xi * dx = r>(ix) + r<(ix)
684  xi += 1.0; // xi = real(ix)
685  }
686  else
687  ys += yp[patch];
688  }
689  counter.advance();
690  }
691 
692  if (ntheta == 0) // (W/eV); (W/sr/eV) for ntheta == 0
693  {
694  if (symmetry == "none")
695  ys *= da;
696  else if (symmetry == "spherical")
697  ys *= cnst::TWO_PI * dx * dx; // Eq.(2)
698  }
699 
700  #ifndef MPI
701  yst += ys * dt; // (J/eV); (J/sr/eV) for ntheta == 0
702  #endif
703 
704  std::string cname(dname + "-ys_" + time_fname(it, ntd));
705  std::string fname(path + cname + ".txt");
706  std::string header(cname + "\n" + time_string(it, ntd, t));
707 
708  ArrDbl ybroad(nhv);
709  utils::convolution(fwhm, hv, ys, hv, ybroad);
710  if (gol.get_analysis())
711  {
712  if (ntheta == 0)
713  header += "data in W/sr/eV";
714  else
715  header += "data in W/eV";
716 
717  gol.fit_eval(cname, ybroad);
718  }
719  else
720  {
721  double scale = gol.get_best_scale(cname);
722  if (utils::sign_eqt(scale - 1.0, cnst::SCALE_EQT) == 0)
723  {
724  if (ntheta == 0)
725  header += "data in W/sr/eV";
726  else
727  header += "data in W/eV";
728  }
729  else
730  {
731  header += "data in arbitrary_units";
732  ybroad *= scale;
733  }
734  ybroad.to_file(fname, header);
735  }
736 }

738 //-----------------------------------------------------------------------------
739 
740 void Detector::evaluate(const std::vector<size_t> &ndim,
741  const Grid &g, const Mesh &m, const Database &d,
742  const Table &tbl, const size_t jt, const double t,
743  const double dt, const int ntd, Goal &gol)
744 {
745  if (ntheta != 0)
746  { // only parallel Rays are allowed in analysis with spherical symmetry
747  std::cerr << "Error: ntheta is not zero in Detector.evaluate()"
748  << std::endl;
749  exit(EXIT_FAILURE);
750  }
751  size_t izone = utils::one_to_two(ndim, jt).first;
752 
753  if (izone == Zone::BOUNDING_ZONE)
754  {
755  // done searching; retrieve best icase for each Zone
756  size_t nzones = ndim.size();
757  std::vector<size_t> indx(nzones);
758  size_t icase;
759  std::pair<size_t, size_t> pr;
760  for (size_t iz = 1; iz < nzones; ++iz)
761  {
762  icase = gol.get_objective(iz-1)->get_best_case();
763  pr = utils::one_to_two(ndim, icase);
764  if (pr.first != iz)
765  {
766  std::cerr << "Error in Detector::evaluate: "
767  << "Zone ID mismatch: " << pr.first << " " << iz
768  << std::endl;
769  exit(EXIT_FAILURE);
770  }
771  indx.at(iz) = pr.second;
772  }
773 
774  // no Rays are needed through the bounding Zone (#0), as
775  // it only has this one and only case (vacuum, typically)
776  indx.at(Zone::BOUNDING_ZONE) = 0;
777 
778  // compute product-based index it for the best fits from indx and ndim,
779  // as if it had come from a search with symmetry == "none"
780  gol.set_best_case(utils::many_to_one(ndim, indx));
781  }
782  else // evaluate the Ray through izone, which is not the bounding Zone
783  {
784  size_t iz = izone - 1; // innermost Zone is #1 (#0 is bounding Zone)
785  IntPair patch(IntPair(iz, 0)); // only one Ray per Zone for now
786  do_patch(NULL, patch, g, m, d, tbl, jt, t, dt, ntd, true, gol);
787  }
788 }
