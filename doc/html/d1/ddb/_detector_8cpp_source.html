408 void space_integ(const PatchSpectrum &pout)
409 {
410  IntPair patch(pout.ix, pout.iy);
411 
412  if (this_det->symmetry == "none")
413  {
414  for (size_t ihv = 0; ihv < this_det->nhv; ++ihv)
415  {
416  this_det->ys[ihv] += pout.yp[ihv]; // (W/eV); (W/cm2/sr/eV) for ntheta == 0
417  this_det->yt[patch][ihv] += pout.yp[ihv] * dt; // (J/eV); (J/cm2/sr/eV) for ntheta == 0
418  }
419  }
420  else if (this_det->symmetry == "spherical") // Eq.(2)
421  {
422  if (this_det->ntheta == 0)
423  {
424  double xi = static_cast<double>(pout.ix);
425  if (pout.ix == 0) // dx = diameter of the central circle
426  for (size_t ihv = 0; ihv < this_det->nhv; ++ihv)
427  this_det->ys[ihv] += pout.yp[ihv] / 8.0;
428  else // ring width = r>(ix) - r<(ix) = dx, for ix > 0
429  for (size_t ihv = 0; ihv < this_det->nhv; ++ihv)
430  this_det->ys[ihv] += pout.yp[ihv] * xi; // xi * dx = r>(ix) + r<(ix)
431  }
432  else
433  for (size_t ihv = 0; ihv < this_det->nhv; ++ihv)
434  this_det->ys[ihv] += pout.yp[ihv];
435  }
436  parent->advance();
437  ++j;
438  if (parent != nullptr) if (j == parent->get_freq()) {j = 0;}
439 }

970 void Detector::do_Ray(Progress *parent,
971  const IntPair &patch, const IntPair &direction,
972  const Grid &g, const Mesh &m, const Database &d,
973  const Table &tbl,
974  const std::string &froot, const std::string &hroot,
975  const std::string &tname, Goal &gol)
976 {

978  const size_t &iy = patch.second;
979 
980  Vector3d r0; // where the Ray hits this patch on *this Detector
981  if (symmetry == "none")
982  r0 = ro + ux * ix + uy * iy;
983  else if (symmetry == "spherical")
984  r0 = rc + ux * ix;
985 
986  double theta = static_cast<double>(direction.first) * dtheta;
987  double phi = static_cast<double>(direction.second) * dphi;
988  Vector3d cvec(local_to_global(-cnst::CV * Vector3d(theta, phi)));
989 
990  const double EQT = 1.0e-15;
991  auto f = m.get_zone(Zone::BOUNDING_ZONE)->get_face(0);
992  FaceID fid(Zone::BOUNDING_ZONE, -3);
993  RetIntercept pt = f->intercept(g, r0, cvec, EQT, fid);
994 
995  // set up Progress counter
996  int next_level;
997  size_t next_freq;
998  if (parent == nullptr)
999  {
1000  next_level = 0;
1001  next_freq = 0;
1002  }
1003  else
1004  {
1005  next_level = parent->get_level() + 1;
1006  next_freq = parent->get_next_freq(freq_trace);
1007  }
1008 
1009  std::string cname(froot + omega_fname(direction) + "_" + tname);
1010  std::string fname(path + cname);
1011  std::string header(cname + "\n" + hroot + omega_string(direction));
1012  Ray ray(next_level, next_freq, nhv, jmin, jmax, tracking, r0, cvec,
1013  gol.get_analysis(), fname, header);
1014  fname += ".txt";
1015 
1016  if (pt.is_found)
1017  {
1018  ray.diag_id = my_id;
1019  ray.patch_id = patch;
1020  ray.bundle_id = direction;
1021  ray.trace(g, m);
1022  ray.set_backlighter(yback);
1023  ray.cross_Mesh(m, d, tbl, symmetry, ix);

1025  yp[patch] = ray.y;
1026  else // bundle of Rays per patch (W/cm2/eV)

1028  double domega; // pmh_2016_1013
1029  if (direction == INT_PAIR_00) // spherical cap
1030  domega = cnst::FOUR_PI * pow(sin(dtheta/4.0), 2);
1031  else // spherical rectangle
1032  domega = 2.0 * dphi * sin(theta) * sin(dtheta2);
1033  yp[patch] += ray.y * (domega * ez.cos_angle(ray.v));
1034  }
1035  }
1036 
1037  ArrDbl ybroad(nhv);
1038  utils::convolution(fwhm, hv, ray.y, hv, ybroad);
1039  if (gol.get_analysis())
1040  {
1041  header += "\ndata in W/cm2/sr/eV";
1042  gol.fit_eval(cname, ybroad);
1043  }
1044  else
1045  {
1046  double scale = gol.get_best_scale(cname);
1047  if (utils::sign_eqt(scale - 1.0, cnst::SCALE_EQT) == 0)
1048  header += "\ndata in W/cm2/sr/eV";
1049  else
1050  {
1051  header += "\ndata in arbitrary_units";
1052  ybroad *= scale;
1053  }
1054  ybroad.to_file(fname, header);
1055  }
