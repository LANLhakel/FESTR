110double Face::abs_diff(const Face &o) const
111{
112 const double zdiff = static_cast<double>(my_zone)
113 - static_cast<double>(o.my_zone);
114 const double idiff = static_cast<double>(my_id)
115 - static_cast<double>(o.my_id);
116 size_t range = node.size();
117 const size_t f_range = o.node.size();
118 const double rdiff = static_cast<double>(range)
119 - static_cast<double>(f_range);
120
121 if (f_range < range) range = f_range;
122 double d;
123 double diff = 0.0;
124 for (size_t i = 0; i < range; ++i)
125 {
126 d = static_cast<double>(node.at(i) - o.node.at(i));
127 diff += d*d;
128 }
129 return diff + zdiff*zdiff + idiff*idiff + rdiff*rdiff;
130}

155void Face::load_face(std::ifstream &istr, const size_t nnodes)
156{
157 Face::clear(); // derived classes have their own clear() functions
158 istr >> my_zone >> my_id; // load FaceID
159 size_t i, j, nnbr;
160 short int k;
161
162 for (i = 0; i < nnodes; ++i) // load Nodes
163 {
164 istr >> j;
165 add_node(j);
166 }
167
168 utils::find_word(istr, "neighbors");
169 istr >> nnbr;
170 for (i = 0; i < nnbr; ++i) // load Neighbors' FaceID's
171 {
172 istr >> j >> k;
173 add_neighbor(j, k);
174 }
175}
