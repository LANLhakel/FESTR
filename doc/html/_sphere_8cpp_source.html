184Vector3d Sphere::subpoint(const Grid &g, const Vector3d &w) const
185{
186 const Vector3d center = POINT(0);
187 const Vector3d d = w - center;
188 const double dnorm = d.norm();
189
190 if (fabs(dnorm) < Vector3d::get_small())
191 { // w == center; no solution
192 const double BIG = -Vector3d::get_big();
193 return Vector3d(BIG, BIG, BIG);
194 }
195 else // w != center; well defined solution
196 return center + d/dnorm * getr();
197}
