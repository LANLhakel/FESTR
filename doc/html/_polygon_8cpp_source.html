75// *this Polygon is assumed to be convex
76Vector3d Polygon::area2_normal_center(const Grid &g, Vector3d &c) const
77{
78 const Vector3d origin = POINT(0);
79 Vector3d p_next = POINT(1);
80 Vector3d v_next = p_next - origin;
81 Vector3d p_prev, v_prev, s;
82 c.set0();
83 size_t n = size();
84 for (size_t i = 2; i < n; ++i)
85 {
86 p_prev = p_next;
87 p_next = POINT(i);
88
89 // normal vector part
90 v_prev = v_next;
91 v_next = p_next - origin;
92 Vector3d a = v_prev % v_next;
93 s += a;
94
95 // center part
96 Vector3d ctr = (origin + p_prev + p_next) / 3.0; // triangle's centroid
97 c += ctr * a.norm(); // building triangle-area-weighted average vector
98 }
99 c /= s.norm();
100 return s; // |s| = 2*area
101}

150// *this Polygon is assumed to be convex
151// z component of v is assumed == 0 after transformation into *this plane
152bool Polygon::contains(const Grid &g, const Vector3d &w) const
153{ // pmh_2014_1125
154 const int NOT_INITIALIZED = -2;
155 int first_turn = NOT_INITIALIZED;
156 int turn;
157
158 // build local coordinate system in the plane of *this
159 const Vector3d Zhat = normal(g);
160 const Vector3d origin = POINT(0);
161 const Vector3d Xhat = (POINT(1) - origin).normalize();
162 const Vector3d Yhat = Zhat % Xhat;
163
164 // build q as the representation of v in this local coordinate system
165 const Vector3d p = w - origin;
166 const Vector3d q(p*Xhat, p*Yhat, 0.0);
167 Vector3d head3d, tail3d, head, tail;
168
169 // loop over *this Polygon's sides
170 size_t i;
171 const size_t n = size();
172 for (size_t j = 0; j < n; ++j)
173 { // see if q is on the same side relative to each side's vector
174 i = j + 1;
175 if (i == n) i = 0;
176 head3d = POINT(i) - origin;
177 tail3d = POINT(j) - origin;
178 head = Vector3d(head3d*Xhat, head3d*Yhat, 0.0);
179 tail = Vector3d(tail3d*Xhat, tail3d*Yhat, 0.0);
180 turn = q.get_turn(tail, head);
181
182 if (turn != 0)
183 { // points lying exactly on the line are not considered
184 if (first_turn == NOT_INITIALIZED)
185 {
186 first_turn = turn;
187 }
188 else
189 { // if turn differs from first_turn => v is outside of *this
190 if (first_turn != turn) return false;
191 }
192 }
193 }
194
195 // all turns were in the same direction => *this contains v
196 return true;
197}

201RetIntercept Polygon::intercept(const Grid &g, const Vector3d &p,
202 const Vector3d &u, const double eqt,
203 const FaceID &fid) const
204{ // pmh_2014_1125
205 RetIntercept rv;
206 rv.fid = FaceID(get_my_zone(), get_my_id());
207 const Vector3d n = normal(g);
208 const double denominator = u * n;
209
210 if (rv.fid == fid || fabs(denominator) < Vector3d::get_small())
211 {
212 const double BIG = -Vector3d::get_big();
213 rv.t = BIG;
214 rv.w = Vector3d(BIG, BIG, BIG);
215 rv.is_found = false;
216 }
217 else
218 {
219 const Vector3d a = POINT(0);
220 const double numerator = (a - p) * n;
221 rv.t = numerator / denominator; // Eq.(3)
222 rv.w = p + u * rv.t; // Eq.(2)
223 rv.is_found = utils::sign_eqt(rv.t, eqt) == 1 && contains(g, rv.w);
224 }
225
226 return rv;
227}

231Vector3d Polygon::velocity(const Grid &g, const Vector3d &w) const
232{
233 const double SMALL = Vector3d::get_small();
234 Vector3d u;
235 double d;
236 double s = 0.0;
237 for (size_t i = 0; i < size(); ++i)
238 {
239 d = (POINT(i) - w).norm();
240 if (d < SMALL) return VELOCITY(i);
241 d = 1.0 / d;
242 s += d;
243 u += d * VELOCITY(i);
244 }
245 return u / s;
246}
