307void do_patch(const PatchID &pid, PatchSpectrum &pout)
308{
309 IntPair patch(pid.ix, pid.iy);
310 pout.ix = pid.ix;
311 pout.iy = pid.iy;
312
313 // set up Progress counter
314 int next_level;
315 size_t next_freq;
316 if (parent == nullptr)
317 {
318 next_level = 0;
319 next_freq = 0;
320 }
321 else
322 {
323 next_level = parent->get_level() + 1;
324
325 #ifdef MPI
326 size_t pf = parent->get_freq();
327 if (pf == 0)
328 next_freq = 0;
329 else
330 next_freq = ( (pid.j+1) == pf ? this_det->freq_trace : 0 );
331
332 #else
333 next_freq = parent->get_next_freq(this_det->freq_Ray);
334 #endif
335 }
336
337 const std::string SEP("----------------------------------------\n");
338 size_t ncount(this_det->ntheta == 0 ? 1 : (this_det->ntheta-1)*this_det->nphi + 1);
339 Progress counter(this_det->dname + "_Ray", next_level, ncount, next_freq,
340 SEP, std::cout);
341
342 this_det->yp[patch].fill(0.0);
343 std::string cname(this_det->dname + "-yp" + this_det->patch_fname(patch));
344 std::string fname(this_det->path + cname);
345 std::string tname(time_fname(it, ntd));
346 std::string header(time_string(it, ntd, t));
347 header += this_det->patch_string(patch);
348 this_det->do_Ray(&counter, patch, INT_PAIR_00, *g, *m, *d, *tbl,
349 cname, header, tname, *gol);
350 counter.advance();
351
352/*
353 #ifdef OMP
354 omp_set_num_threads(atoi(getenv("OMP_NUM_THREADS")));
355 #pragma omp parallel for default(none) \
356 firstprivate(this_det,gol,tname,header,cname,tbl,d,m,g,patch) shared(counter) collapse(2)
357 #endif
358*/
359 for (size_t itheta = 1; itheta < this_det->ntheta; ++itheta)
360 for (size_t iphi = 0; iphi < this_det->nphi; ++iphi)
361 {
362 this_det->do_Ray(&counter, patch, IntPair(itheta, iphi), *g, *m, *d, *tbl,
363 cname, header, tname, *gol);
364 counter.advance();
365 }
366
367 if (this_det->ntheta > 0) // (W/eV); (W/cm2/sr/eV) for ntheta == 0
368 {
369 if (this_det->symmetry == "none")
370 this_det->yp[patch] *= this_det->da;
371 else if (this_det->symmetry == "spherical") // Eq.(2)
372 {
373 std::cerr << "Spherical symmetry is not available in MPI parallel runs" << std::endl;
374 exit(EXIT_FAILURE);
375 }
376 }
377
378 header = cname + "_" + tname + "\n" + header;
379 fname += "_" + tname + ".txt";
380
381 ArrDbl ybroad(this_det->nhv);
382 utils::convolution(this_det->fwhm, this_det->hv, this_det->yp[patch], this_det->hv, ybroad);
383
384 double scale = gol->get_best_scale(cname);
385 if (utils::sign_eqt(scale - 1.0, cnst::SCALE_EQT) == 0)
386 {
387 if (this_det->ntheta == 0)
388 header += "data in W/cm2/sr/eV";
389 else
390 header += "data in W/eV";
391 }
392 else
393 {
394 header += "data in arbitrary_units";
395 ybroad *= scale;
396 }
397 ybroad.to_file(fname, header);
398
399 pout.nhv = this_det->nhv;
400 pout.yp.resize(pout.nhv);
401 for (size_t ihv = 0; ihv < pout.nhv; ++ihv)
402 pout.yp[ihv] = this_det->yp[patch][ihv];
403
404} // end PatchEnvironment::do_patch

408void space_integ(const PatchSpectrum &pout)
409{
410 IntPair patch(pout.ix, pout.iy);
411
412 if (this_det->symmetry == "none")
413 {
414 for (size_t ihv = 0; ihv < this_det->nhv; ++ihv)
415 {
416 this_det->ys[ihv] += pout.yp[ihv]; // (W/eV); (W/cm2/sr/eV) for ntheta == 0
417 this_det->yt[patch][ihv] += pout.yp[ihv] * dt; // (J/eV); (J/cm2/sr/eV) for ntheta == 0
418 }
419 }
420 else if (this_det->symmetry == "spherical") // Eq.(2)
421 {
422 if (this_det->ntheta == 0)
423 {
424 double xi = static_cast<double>(pout.ix);
425 if (pout.ix == 0) // dx = diameter of the central circle
426 for (size_t ihv = 0; ihv < this_det->nhv; ++ihv)
427 this_det->ys[ihv] += pout.yp[ihv] / 8.0;
428 else // ring width = r>(ix) - r<(ix) = dx, for ix > 0
429 for (size_t ihv = 0; ihv < this_det->nhv; ++ihv)
430 this_det->ys[ihv] += pout.yp[ihv] * xi; // xi * dx = r>(ix) + r<(ix)
431 }
432 else
433 for (size_t ihv = 0; ihv < this_det->nhv; ++ihv)
434 this_det->ys[ihv] += pout.yp[ihv];
435 }
436 parent->advance();
437 ++j;
438 if (parent != nullptr) if (j == parent->get_freq()) {j = 0;}
439}

460Detector::Detector(const size_t freq_patch_in, const size_t freq_Ray_in,
461 const size_t freq_trace_in, const std::string &diag_path,
462 const std::string &dname_in, const std::string &path_in,
463 const std::string &dbase_path, const std::string &sym_in,
464 const int my_id_in, const Vector3d & rc_in,
465 const Vector3d &rx_in, const Vector3d &ry_in,
466 const double dx_in, const double dy_in,
467 const size_t nhv_in,
468 const double hv_min, const double hv_max,
469 const double fwhm_in, const Vector3d &pc_in,
470 const std::string &back_type_in,
471 const std::string &back_value_in, const bool tracking_in,
472 const bool write_Ray_in):
473 yt(), yst(),
474 freq_patch(freq_patch_in), freq_Ray(freq_Ray_in), freq_trace(freq_trace_in),
475 dname(dname_in), path(path_in), symmetry(sym_in), my_id(my_id_in),
476 rc(rc_in), ro(), rx(rx_in), ry(ry_in), ex(), ey(), ez(), bx(), by(), bz(),
477 xr(), yr(), zr(), dx(dx_in), dy(dy_in), da(dx_in*dy_in),
478 ux(), uy(), nhv(0), hv(), hvmin(hv_min), hvmax(hv_max),
479 jmin(nhv_in), jmax(0), fwhm(fwhm_in), back_type(back_type_in),
480 back_fname(""), back_value(-9.0), yback(), tracking(tracking_in),
481 write_Ray(write_Ray_in), nx(0), ny(0), nxd(0), nyd(0), pc(pc_in),
482 theta_max(0.0), ntheta(0), nphi(0), nthetad(0), nphid(0), dtheta(0.0),
483 dphi(0.0), dtheta2(0.0), gdet(), p(), yp(), ys()
484{
485 // set hv grid
486 std::string hvpath(dbase_path + "grids/hv_grid.txt");
487 std::ifstream infile(hvpath.c_str());
488 if (!infile.is_open())
489 {
490 std::cerr << "Error: file " << hvpath << " is not open in "
491 << "Detector::Detector(parametrized)" << std::endl;
492 exit(EXIT_FAILURE);
493 }
494 utils::find_line(infile, "Grid points:");
495 for (size_t ihv = 0; ihv < nhv_in; ++ihv)
496 {
497 size_t j;
498 double e;
499 infile >> j >> e;
500 bool min_true(e >= hvmin);
501 bool max_true(e <= hvmax);
502 if (min_true && max_true) hv.emplace_back(std::move(e));
503 if (min_true && ihv < jmin) jmin = ihv;
504 if (max_true && ihv > jmax) jmax = ihv;
505 }
506 infile.close();
507 infile.clear();
508 if (hv.empty())
509 {
510 std::cerr << "Error: The hv range of Detector " << dname << " has no "
511 << "overlap with the hv range of the Database" << std::endl;
512 exit(EXIT_FAILURE);
513 }
514
515 // set patch discretization
516 nx = static_cast<size_t>(utils::nint( 2.0 * rx_in.norm() / dx_in ));
517 ny = static_cast<size_t>(utils::nint( 2.0 * ry_in.norm() / dy_in ));
518 if (sym_in == "spherical")
519 {
520 nx = nx / 2 + 1;
521 dx = rx_in.norm() / (static_cast<double>(nx) - 0.5); // Eq.(1)
522 ny = 1; // unused
523 dy = 2.0 * ry_in.norm(); // unused
524 // unlike for sym_in == "none", da != const, and hence (* da) cannot be
525 // factored when calculating space-integrated spectrum in do_patches()
526 da = 1.0;
527 }
528 nxd = utils::ndigits(nx);
529 nyd = utils::ndigits(ny);
530
531 // initialize space and time integrated spectral arrays
532 nhv = jmax - jmin + 1;
533
534 //#ifndef MPI
535 ArrDbl y0;
536 y0.assign(nhv, 0.0);
537 //#endif
538
539 for (size_t iy = 0; iy < ny; ++iy)
540 for (size_t ix = 0; ix < nx; ++ix)
541 {
542 IntPair patch(ix, iy);
543 //#ifndef MPI
544 yt.insert(SpecPatch::value_type(patch, y0));
545 //#endif
546 yp[patch].assign(nhv, 0.0);
547 }
548 ys.assign(nhv, 0.0);
549 yst.assign(nhv, 0.0);
550
551 // set patch vectors
552 ex = rx.unit();
553 ey = ry.unit();
554 ez = ex % ey;
555 ux = dx * ex;
556 uy = dy * ey;
557
558 // orient the basis for Ray bundles
559 bz = (rc - pc).unit();
560 bx = ex.perpendicular_to(bz);
561 by = ey.perpendicular_to(bz);
562 if ( bx.norm() > by.norm() )
563 {
564 bx.normalize();
565 by = bz % bx;
566 }
567 else
568 {
569 by.normalize();
570 bx = by % bz;
571 }
572 // set the local_to_global() transformation matrix
573 xr = Vector3d(bx.getx(), by.getx(), bz.getx()); // x-row
574 yr = Vector3d(bx.gety(), by.gety(), bz.gety()); // y-row
575 zr = Vector3d(bx.getz(), by.getz(), bz.getz()); // z-row
576
577 // define Polygon's vertices
578 Vector3d r(rc - rx - ry);
579 gdet.add_node(Node(0, r)); // lower-left corner (Polygon origin)
580 ro = r + (ux + uy) / 2.0; // lower-left corner (origin for Rays)
581 gdet.add_node(Node(1, rc + rx - ry)); // lower-right corner
582 gdet.add_node(Node(2, rc + rx + ry)); // upper-right corner
583 gdet.add_node(Node(3, rc - rx + ry)); // upper-left corner
584 for (size_t i = 0; i < NPT; ++i) p.add_node(i);
585
586 // set up the backlighter
587 if (back_type == "file")
588 {
589 back_fname = back_value_in;
590 back_value = -5.0;
591 }
592 else
593 {
594 back_value = atof(back_value_in.c_str());
595 }
596
597 yback.assign(nhv, back_value); // treating back_type == "flat" as default
598 if (back_type == "file")
599 {
600 std::string back_path = diag_path + back_fname;
601 std::ifstream back_file(back_path.c_str());
602 if (!back_file.is_open())
603 {
604 std::cerr << "Error: file " << back_fname << " is not open in "
605 << "Detector::Detector(parametrized)" << std::endl;
606 exit(EXIT_FAILURE);
607 }
608 utils::find_word(back_file, "nhv:");
609 size_t nback;
610 back_file >> nback;
611 std::vector<double> x(nback), y(nback); // the given backlighter table
612 utils::find_line(back_file, "# hv(eV) I0(W/cm2/sr/eV)");
613 for (size_t i = 0; i < nback; ++i) back_file >> x.at(i) >> y.at(i);
614 back_file.close();
615 back_file.clear();
616 utils::syngrids(x, y, nback, "lin", "lin", hv, yback, nhv);
617 }
618 else if (back_type == "blackbody")
619 {
620 for (size_t ihv = 0; ihv < nhv; ++ihv)
621 yback.at(ihv) = utils::planckian(hv.at(ihv), back_value);
622 }
623
624 // write out the working backlighter spectrum
625 hvpath = path_in + dname_in + "-hv_grid.txt";
626 std::ofstream out_hv(hvpath.c_str());
627 if (!out_hv.is_open())
628 {
629 std::cerr << "Error: file " << hvpath << " is not open." << std::endl;
630 exit(EXIT_FAILURE);
631 }
632 out_hv << "\n Number of bits:\n 0\n\n Number of grid points:\n"
633 << utils::int_to_string(nhv, ' ', cnst::INT_WIDTH+1)
634 << "\n\n Grid points:\n";
635 std::string backpath = path_in + dname_in + "-backlighter.txt";
636 std::ofstream out_back(backpath.c_str());
637 if (!out_back.is_open())
638 {
639 std::cerr << "Error: file " << backpath << " is not open." << std::endl;
640 exit(EXIT_FAILURE);
641 }
642 out_back << dname_in + "-backlighter\ndata in W/cm2/sr/eV" << std::endl;
643 for (size_t ihv = 0; ihv < nhv; ++ihv)
644 {
645 out_hv << utils::int_to_string(ihv, ' ', cnst::INT_WIDTH+1) << " "
646 << utils::double_to_string(hv.at(ihv)) << std::endl;
647 out_back << utils::double_to_string(yback.at(ihv)) << std::endl;
648 }
649 out_hv.close();
650 out_hv.clear();
651 out_back.close();
652 out_back.clear();
653}

973void Detector::do_Ray(Progress *parent,
974 const IntPair &patch, const IntPair &direction,
975 const Grid &g, const Mesh &m, const Database &d,
976 const Table &tbl,
977 const std::string &froot, const std::string &hroot,
978 const std::string &tname, Goal &gol)
979{
980 const size_t &ix = patch.first;
981 const size_t &iy = patch.second;
982
983 Vector3d r0; // where the Ray hits this patch on *this Detector
984 if (symmetry == "none")
985 r0 = ro + ux * ix + uy * iy;
986 else if (symmetry == "spherical")
987 r0 = rc + ux * ix;
988
989 double theta = static_cast<double>(direction.first) * dtheta;
990 double phi = static_cast<double>(direction.second) * dphi;
991 Vector3d cvec(local_to_global(-cnst::CV * Vector3d(theta, phi)));
992
993 const double EQT = 1.0e-15;
994 auto f = m.get_zone(Zone::BOUNDING_ZONE)->get_face(0);
995 FaceID fid(Zone::BOUNDING_ZONE, -3);
996 RetIntercept pt = f->intercept(g, r0, cvec, EQT, fid);
997
998 // set up Progress counter
999 int next_level;
1000 size_t next_freq;
1001 if (parent == nullptr)
1002 {
1003 next_level = 0;
1004 next_freq = 0;
1005 }
1006 else
1007 {
1008 next_level = parent->get_level() + 1;
1009 next_freq = parent->get_next_freq(freq_trace);
1010 }
1011
1012 std::string cname(froot + omega_fname(direction) + "_" + tname);
1013 std::string fname(path + cname);
1014 std::string header(cname + "\n" + hroot + omega_string(direction));
1015 Ray ray(next_level, next_freq, nhv, jmin, jmax, tracking, r0, cvec,
1016 gol.get_analysis(), fname, header);
1017 fname += ".txt";
1018
1019 if (pt.is_found)
1020 {
1021 ray.diag_id = my_id;
1022 ray.patch_id = patch;
1023 ray.bundle_id = direction;
1024 ray.trace(g, m);
1025 ray.set_backlighter(yback);
1026 ray.cross_Mesh(m, d, tbl, symmetry, ix);
1027 if (ntheta == 0) // parallel Rays; one Ray per patch (W/cm2/sr/eV)
1028 yp[patch] = ray.y;
1029 else // bundle of Rays per patch (W/cm2/eV)
1030 {
1031 double domega; // pmh_2016_1013
1032 if (direction == INT_PAIR_00) // spherical cap
1033 domega = cnst::FOUR_PI * pow(sin(dtheta/4.0), 2);
1034 else // spherical rectangle
1035 domega = 2.0 * dphi * sin(theta) * sin(dtheta2);
1036 yp[patch] += ray.y * (domega * ez.cos_angle(ray.v));
1037 }
1038 }
1039
1040 ArrDbl ybroad(nhv);
1041 utils::convolution(fwhm, hv, ray.y, hv, ybroad);
1042 if (gol.get_analysis())
1043 {
1044 header += "\ndata in W/cm2/sr/eV";
1045 gol.fit_eval(cname, ybroad);
1046 }
1047 else
1048 {
1049 if (write_Ray)
1050 {
1051 double scale = gol.get_best_scale(cname);
1052 if (utils::sign_eqt(scale - 1.0, cnst::SCALE_EQT) == 0)
1053 header += "\ndata in W/cm2/sr/eV";
1054 else
1055 {
1056 header += "\ndata in arbitrary_units";
1057 ybroad *= scale;
1058 }
1059 ybroad.to_file(fname, header);
1060 }
1061 }
1062}

1066void Detector::do_patch(Progress *parent, const IntPair &patch,
1067 const Grid &g, const Mesh &m, const Database &d,
1068 const Table &tbl,
1069 const size_t it, const double t, const double dt,
1070 const int ntd, const bool peel_onion, Goal &gol)
1071{
1072 // set up Progress counter
1073 int next_level;
1074 size_t next_freq;
1075 if (parent == nullptr)
1076 {
1077 next_level = 0;
1078 next_freq = 0;
1079 }
1080 else
1081 {
1082 next_level = parent->get_level() + 1;
1083 next_freq = parent->get_next_freq(freq_Ray);
1084 }
1085 const std::string SEP("----------------------------------------\n");
1086 size_t ncount(ntheta == 0 ? 1 : (ntheta-1)*nphi + 1);
1087 Progress counter(dname + "_Ray", next_level, ncount, next_freq,
1088 SEP, std::cout);
1089
1090 yp[patch].fill(0.0);
1091 std::string cname(dname + "-yp" + patch_fname(patch));
1092 std::string fname(path + cname);
1093 std::string tname(time_fname(it, ntd));
1094 std::string header(time_string(it, ntd, t));
1095 header += patch_string(patch);
1096 do_Ray(&counter, patch, INT_PAIR_00, g, m, d, tbl,
1097 cname, header, tname, gol);
1098 counter.advance();
1099
1100 for (size_t itheta = 1; itheta < ntheta; ++itheta)
1101 for (size_t iphi = 0; iphi < nphi; ++iphi)
1102 {
1103 do_Ray(&counter, patch, IntPair(itheta, iphi), g, m, d, tbl,
1104 cname, header, tname, gol);
1105 counter.advance();
1106 }
1107
1108 if (ntheta > 0) // (W/eV); (W/cm2/sr/eV) for ntheta == 0
1109 {
1110 if (symmetry == "none")
1111 yp[patch] *= da;
1112 else if (symmetry == "spherical") // Eq.(2)
1113 {
1114 double darea;
1115 if (patch.first == 0) // ix == 0 (central circle)
1116 darea = cnst::PI * pow(dx/2.0, 2);
1117 else // ix > 0 (circular strip of width dx)
1118 darea = cnst::TWO_PI * static_cast<double>(patch.first) * pow(dx, 2);
1119 yp[patch] *= darea;
1120 }
1121 }
1122
1123 #ifndef MPI
1124 yt[patch] += yp[patch] * dt; // (J/eV); (J/cm2/sr/eV) for ntheta == 0
1125 #endif
1126
1127 header = cname + "_" + tname + "\n" + header;
1128 fname += "_" + tname + ".txt";
1129
1130 ArrDbl ybroad(nhv);
1131 utils::convolution(fwhm, hv, yp[patch], hv, ybroad);
1132 if (gol.get_analysis())
1133 {
1134 if (ntheta == 0)
1135 header += "data in W/cm2/sr/eV";
1136 else
1137 header += "data in W/eV";
1138
1139 if (peel_onion)
1140 {
1141 // one Ray per Zone for now
1142 auto obj = gol.get_objective(patch.first);
1143 obj->update_best(ybroad, nhv, it);
1144 }
1145 else
1146 gol.fit_eval(cname, ybroad);
1147 }
1148 else // postprocessing
1149 {
1150 double scale = gol.get_best_scale(cname);
1151 if (utils::sign_eqt(scale - 1.0, cnst::SCALE_EQT) == 0)
1152 {
1153 if (ntheta == 0)
1154 header += "data in W/cm2/sr/eV";
1155 else
1156 header += "data in W/eV";
1157 }
1158 else
1159 {
1160 header += "data in arbitrary_units";
1161 ybroad *= scale;
1162 }
1163 ybroad.to_file(fname, header);
1164 }
1165}

1291void Detector::evaluate(const std::vector<size_t> &ndim,
1292 const Grid &g, const Mesh &m, const Database &d,
1293 const Table &tbl, const size_t jt, const double t,
1294 const double dt, const int ntd, Goal &gol)
1295{
1296 if (ntheta != 0)
1297 { // only parallel Rays are allowed in analysis with spherical symmetry
1298 std::cerr << "Error: ntheta is not zero in Detector.evaluate()"
1299 << std::endl;
1300 exit(EXIT_FAILURE);
1301 }
1302 size_t izone = utils::one_to_two(ndim, jt).first;
1303
1304 if (izone == Zone::BOUNDING_ZONE)
1305 {
1306 // done searching; retrieve best icase for each Zone
1307 size_t nzones = ndim.size();
1308 std::vector<size_t> indx(nzones);
1309 size_t icase;
1310 std::pair<size_t, size_t> pr;
1311 for (size_t iz = 1; iz < nzones; ++iz)
1312 {
1313 icase = gol.get_objective(iz-1)->get_best_case();
1314 pr = utils::one_to_two(ndim, icase);
1315 if (pr.first != iz)
1316 {
1317 std::cerr << "Error in Detector::evaluate: "
1318 << "Zone ID mismatch: " << pr.first << " " << iz
1319 << std::endl;
1320 exit(EXIT_FAILURE);
1321 }
1322 indx.at(iz) = pr.second;
1323 }
1324
1325 // no Rays are needed through the bounding Zone (#0), as
1326 // it only has this one and only case (vacuum, typically)
1327 indx.at(Zone::BOUNDING_ZONE) = 0;
1328
1329 // compute product-based index it for the best fits from indx and ndim,
1330 // as if it had come from a search with symmetry == "none"
1331 gol.set_best_case(utils::many_to_one(ndim, indx));
1332 }
1333 else // evaluate the Ray through izone, which is not the bounding Zone
1334 {
1335 size_t iz = izone - 1; // innermost Zone is #1 (#0 is bounding Zone)
1336 IntPair patch(IntPair(iz, 0)); // only one Ray per Zone for now
1337 do_patch(nullptr, patch, g, m, d, tbl, jt, t, dt, ntd, true, gol);
1338 }
