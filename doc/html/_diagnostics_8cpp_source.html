221void Diagnostics::postprocess(const Database &d, const Hydro &h, Goal &gol)
222{
223 #include <diag_exec.inc>
224 std::string name("Time step");
225
226 { // begin block defining Progress diag_counter scope
227 Progress diag_counter(name, level, nintervals, freq, DIAGSEP, std::cout);
228 for (size_t j = 0; j < nintervals; ++j) // loop over time
229 {
230 size_t it = h.time_index_at(j);
231 double t = h.time_at(it);
232 double dt = h.dt_at(it);
233 h.load_at(it, g, m);
234 Progress det_counter("Detector", level+1, ndet, diag_counter.get_next_freq(freq_det), DETSEP, std::cout);
235 for (size_t id = 0; id < ndet; ++id) // loop over Detectors
236 {
237 det.at(id).do_patches(&det_counter, g, m, d,
238 tbl, it, t, dt, ntd, gol);
239 det_counter.advance();
240 }
241 diag_counter.advance();
242 } // end loop over time
243 } // end block defining Progress diag_counter scope
244
245 #ifdef MPI
246 MPI_Barrier(MPI_COMM_WORLD);
247 int my_rank;
248 MPI_Comm_rank(MPI_COMM_WORLD, &my_rank);
249 if (my_rank == 0)
250 {
251 #endif
252
253 if (freq_det > 0)
254 {
255 std::cout << "\nbegin processing time-integrated, "
256 << "space-resolved spectra ...\n" << std::endl;
257 }
258
259 Progress det_counter("Detector", level+1, ndet, freq_det,
260 DETSEP, std::cout);
261 for (size_t id = 0; id < ndet; ++id) // loop over Detectors
262 {
263 Detector &deti = det.at(id);
264 std::string header(deti.get_dname() + "-yst");

276 double fwhm = deti.get_fwhm();

278 ArrDbl ybroad(nhv);

298void Diagnostics::analyze(const Database &d, Hydro &h, Goal &gol)
299{
300 #include <diag_exec.inc>
301 std::vector<size_t> ndim = h.get_ndim();
302 size_t n = ndim.size();
303 size_t current_zone;
304 size_t previous_zone = n;
305 std::string name("Case");
306
307 { // begin block defining Progress diag_counter scope
308 Progress diag_counter(name, level, nintervals, freq, DIAGSEP, std::cout);
309
310 for (size_t j = 0; j < nintervals; ++j) // loop over cases
311 {
312 size_t it = h.time_index_at(j);
313 double t = h.time_at(0);
314 double dt = h.dt_at(0);
315
316 if (det.at(0).get_symmetry() == "spherical")
317 {
318 size_t jt = nintervals - it - 1; // peeling the onion
319 h.load_at(jt, g, m); // using summation-based index jt
320 if (it == 0)
321 { // set bounding Zone (fixed) on first pass through it loop
322 previous_zone = n - 1;
323 h.load_at(0, g, m);
324 }
325 else // it > 0
326 {
327 current_zone = utils::one_to_two(ndim, jt).first;
328 if (current_zone != previous_zone)
329 {
330 auto obj = gol.get_objective(previous_zone-1);
331 h.load_at(obj->get_best_case(), g, m);
332 previous_zone = current_zone;
333 }
334 }
335 det.at(0).evaluate(ndim, g, m, d, tbl, jt, t, dt, ntd, gol);
336 }
337
338 else // "symmetry == none" path
339 {
340 h.load_at(it, g, m); // using product-based index it
341 Progress det_counter("Detector", level+1, ndet,
342 diag_counter.get_next_freq(freq_det),
343 DETSEP, std::cout);
344 for (size_t id = 0; id < ndet; ++id) // loop over Detectors
345 {
346 det.at(id).do_patches(&det_counter, g, m, d, tbl, it, t, dt,
347 ntd, gol);
348 det_counter.advance();
349 }
350 } // det.at(0).get_symmetry()
351
352 gol.update_best(it);
353 diag_counter.advance();
354 } // end loop over cases
355 } // end block defining Progress diag_counter scope
356
357 // redo the best case and write results
358 gol.set_analysis(false);
359
360 // switching from summation-based indices jt to product-based index it,
361 // as if we hadn't taken advantage of spherical symmetry during search
362 if (h.get_symmetry() == "spherical") h.set_symmetry("none");
363
364 size_t it = gol.get_best_case();
365 h.load_at(it, g, m);
366 double t = h.time_at(0);
367 double dt = h.dt_at(0);
368 ntd = h.get_ntd();
369 for (size_t id = 0; id < ndet; ++id) // loop over Detectors
370 det.at(id).do_patches(nullptr, g, m, d, tbl, it, t, dt, ntd, gol);
371
372 gol.set_analysis(true);
373 size_t nzones = m.size();
374 std::string zone_content;
375 std::string best_case_fname(outpath + "best_case.dat");
376 std::ofstream best_case_plot(best_case_fname.c_str());
377 best_case_fname = outpath + "best_case.txt";
378 std::ofstream best_case_full(best_case_fname.c_str());
379 best_case_full << "time 0 s";
380 std::string header("# Zone ");
381 header += "ne(el./cm3) te(eV) tr(eV) ";
382 header += "np(ions/cm3) fps";
383 std::cout << header;
384 best_case_plot << header;
385 for (size_t iz = 0; iz < nzones; ++iz)
386 {
387 if (iz > 0)
388 {
389 zone_content = "\n"
390 + utils::int_to_string(iz, ' ', cnst::INT_WIDTH)
391 + m.get_zone(iz)->mat_to_string_plot();
392 std::cout << zone_content;
393 best_case_plot << zone_content;
394 }
395 best_case_full << "\n\n" << m.get_zone(iz)->mat_to_string_full();
396 }
397 std::cout << std::endl;
398 best_case_plot << std::endl;
399 best_case_plot.close();
400 best_case_plot.clear();
401 best_case_full << std::endl;
402 best_case_full.close();
403 best_case_full.clear();
404}
