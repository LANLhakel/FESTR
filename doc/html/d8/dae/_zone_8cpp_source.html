227 RetIntercept Zone::hit(const Grid &g, const Vector3d &p, const Vector3d &u,
228  const FaceID &fid) const
229 {
230  // minimum_Zone_size cannot be smaller than ~ Bohr radius (5e-9 cm);
231  // choosing minimum_Zone_size = 1e-8 cm, then:
232  // photon_time_of_flight = minimum_Zone_size / speed_of_light
233  // = (1e-8 cm) / (3e10 cm/s) ~ 3e-19 s >~ EQT = 1e-19
234  const double EQT = 1.0e-19;
235  RetIntercept rv, pt;
236  const size_t n = size();
237 
238  rv.t = Vector3d::get_big();
239  for (size_t i = 0; i < n; ++i) // loop over *this Zone's Faces
240  {
241  pt = face.at(i)->intercept(g, p, u, EQT, fid);
242  if (pt.is_found && pt.t < rv.t) rv = pt;
243  }
244 
245  if (!rv.is_found)
246  { // try again, this time tracing from zp instead of p
247  Vector3d zp = zone_point(g, p);
248  rv.t = Vector3d::get_big();
249  for (size_t i = 0; i < n; ++i) // loop over *this Zone's Faces
250  {
251  pt = face.at(i)->intercept(g, zp, u, EQT, fid);
252  if (pt.is_found && pt.t < rv.t) rv = pt;
253  }
254  }
255 
256  return rv;
257 }

314 void Zone::load_spectra(const Database &d, const Table &tbl,
315  size_t &ite, size_t &itr, size_t &ine,
316  const std::string &symmetry, const size_t ix,
317  const bool analysis,
318  const size_t jmin, const size_t jmax,
319  ArrDbl &em, ArrDbl &ab, ArrDbl &sc) // not const
320 {
321  size_t nhv_max = d.get_nhv();
322  size_t nhv = jmax - jmin + 1;
323  em.assign(nhv, 0.0);
324  ab.assign(nhv, 0.0);
325  sc.assign(nhv, 0.0);
326 
327  if (nmat > 0)
328  {
329  /* NOTE:
330  In analysis mode with spherical symmetry we are "peeling the onion";
331  in that case the ordering of Rays, as they are sent through the Mesh,
332  is reversed from the original outward direction. Hence the storing of
333  optical data is not used in that case, as it relies on shooting the
334  central Ray through the Mesh first (since that Ray is guaranteed
335  to cross all spherical Zones in the Mesh).
336  */
337  if ((symmetry == "none") || analysis ||
338  ((symmetry != "none") && (ix == 0))) // ix == 0 == central Ray
339  {
340  ArrDbl v; // working array used by each material and quantity
341  v.assign(nhv, 0.0); // actually allocate nhv elements
342 
343  NeData ne_data(d.find_ne(tbl,te,ite,tr,itr,np,nmat,mat,fp,ine));
344  ne = ne_data.first;
345  std::string froot, fname;
346  double fpop;
347  const std::string dirpath(d.get_path() + "spectra/");
348  for (unsigned short int i = 0; i < nmat; ++i)
349  {
350  std::string m(tbl.get_F(mat.at(i)));
351  froot = dirpath + m + "/" + m + ne_data.second;
352  fpop = fp.at(i);
353 
354  fname = froot + "em.txt";
355  utils::load_array(fname, nhv_max, jmin, jmax, v);
356  em += fpop * v;
357 
358  fname = froot + "ab.txt";
359  utils::load_array(fname, nhv_max, jmin, jmax, v);
360  ab += fpop * v;
361 
362  fname = froot + "sc.txt";
363  utils::load_array(fname, nhv_max, jmin, jmax, v);
364  sc += fpop * v;
365  }
366  em *= np;
367  ab *= np;
368  sc *= np;
369 
370  if (symmetry != "none") // in 1-D, save spectral data from ix == 0
371  { // Ray for reuse by other Rays within the same time interval
372  emis = em;
373  absp = ab;
374  scat = sc;
375  }
376  }
377  else // (symmetry != "none") && (ix > 0), &&!analysis, retrieve
378  // saved data calculated earlier for the central Ray (ix == 0)
379  {
380  em = emis;
381  ab = absp;
382  sc = scat;
383  }
384 
385  } // nmat > 0
386 }
