165 RetIntercept Zone::hit(const Grid &g, const Vector3d &p, const Vector3d &u,
166  const FaceID &fid) const
167 {
168  // minimum_Zone_size cannot be smaller than ~ Bohr radius (5e-9 cm);
169  // choosing minimum_Zone_size = 1e-8 cm, then:
170  // photon_time_of_flight = minimum_Zone_size / speed_of_light
171  // = (1e-8 cm) / (3e10 cm/s) ~ 3e-19 s >~ EQT = 1e-19
172  const double EQT = 1.0e-19;
173  RetIntercept rv, pt;
174  const size_t n = size();
175 
176  rv.t = Vector3d::get_big();
177  for (size_t i = 0; i < n; ++i) // loop over *this Zone's Faces
178  {
179  pt = face.at(i)->intercept(g, p, u, EQT, fid);
180  if (pt.is_found && pt.t < rv.t) rv = pt;
181  }
182  return rv;
183 }

240 void Zone::load_spectra(const Database &d, const Table &tbl,
241  size_t &ite, size_t &itr, size_t &ine,
242  const std::string &symmetry, const size_t ix,
243  const bool analysis,
244  const size_t jmin, const size_t jmax,
245  ArrDbl &em, ArrDbl &ab, ArrDbl &sc) // not const
246 {
247  size_t nhv_max = d.get_nhv();
248  size_t nhv = jmax - jmin + 1;
249  em.assign(nhv, 0.0);
250  ab.assign(nhv, 0.0);
251  sc.assign(nhv, 0.0);
252 
253  if (nmat > 0)
254  {
255  /* NOTE:
256  In analysis mode with spherical symmetry we are "peeling the onion";
257  in that case the ordering of Rays, as they are sent through the Mesh,
258  is reversed from the original outward direction. Hence the storing of
259  optical data is not used in that case, as it relies on shooting the
260  central Ray through the Mesh first (since that Ray is guaranteed
261  to cross all spherical Zones in the Mesh).
262  */
263  if ((symmetry == "none") || analysis ||
264  ((symmetry != "none") && (ix == 0))) // ix == 0 == central Ray
265  {
266  ArrDbl v; // working array used by each material and quantity
267  v.assign(nhv, 0.0); // actually allocate nhv elements
268 
269  NeData ne_data(d.find_ne(tbl,te,ite,tr,itr,np,nmat,mat,fp,ine));
270  ne = ne_data.first;
271  std::string froot, fname;
272  double fpop;
273  const std::string dirpath(d.get_path() + "spectra/");
274  for (unsigned short int i = 0; i < nmat; ++i)
275  {
276  std::string m(tbl.get_F(mat.at(i)));
277  froot = dirpath + m + "/" + m + ne_data.second;
278  fpop = fp.at(i);
279 
280  fname = froot + "em.txt";
281  utils::load_array(fname, nhv_max, jmin, jmax, v);
282  em += fpop * v;
283 
284  fname = froot + "ab.txt";
285  utils::load_array(fname, nhv_max, jmin, jmax, v);
286  ab += fpop * v;
287 
288  fname = froot + "sc.txt";
289  utils::load_array(fname, nhv_max, jmin, jmax, v);
290  sc += fpop * v;
291  }
292  em *= np;
293  ab *= np;
294  sc *= np;
295 
296  if (symmetry != "none") // in 1-D, save spectral data from ix == 0
297  { // Ray for reuse by other Rays within the same time interval
298  emis = em;
299  absp = ab;
300  scat = sc;
301  }
302  }
303  else // (symmetry != "none") && (ix > 0), &&!analysis, retrieve
304  // saved data calculated earlier for the central Ray (ix == 0)
305  {
306  em = emis;
307  ab = absp;
308  sc = scat;
309  }
310 
311  } // nmat > 0
312 }
