248double Vector3d::cos_angle(const Vector3d &v) const
249{
250 const double tnorm = norm();
251 const double vnorm = v.norm();
252 if (fabs(tnorm) < SMALL || fabs(vnorm) < SMALL) return -4.0;
253 return (*this * v) / (tnorm * vnorm);

350std::pair<bool, Vector3d>
351lines_intersection(const Vector3d &p, const Vector3d &u,
352 const Vector3d &a, const Vector3d &b,
353 const double eqt)
354{ // pmh_2022_0427
355 double zero = fabs(eqt);
356 double big = Vector3d::get_big();
357 std::pair<bool, Vector3d> rv(false, Vector3d(big, big, big));
358 double px = p.getx();
359 double py = p.gety();
360 double un = u.norm();
361 double ux = u.getx() / un;
362 double uy = u.gety() / un;
363 double ra = a.getx();
364 double za = a.gety();
365 double dr = b.getx() - ra;
366 double dz = b.gety() - za;
367 double d = uy * dr - ux * dz;
368 if (fabs(d) < zero) return rv; // parallel lines
369 double ds = ux * (za - py) - uy * (ra - px);
370 double s = ds / d;
371 rv.first = s > -zero && s < 1.0+zero;
372 rv.second = a + (b - a) * s;
373 return rv;
374}
