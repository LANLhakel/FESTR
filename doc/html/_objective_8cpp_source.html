37Objective::Objective(const std::string &oname_in, const std::string &path_in,
38 const int my_id_in, const double weight_in,
39 const bool x_is_present_in, const bool w_is_present_in,
40 const bool rescale_in):
41 oname(oname_in), path(path_in), my_id(my_id_in), weight(weight_in),
42 x_is_present(x_is_present_in), w_is_present(w_is_present_in),
43 x(), y(), w(), best_fitness(-BIG), best_case(0),
44 rescale(rescale_in), best_scale(1.0)
45{
46 std::string fname(path + oname + ".txt");
47 std::ifstream infile(fname.c_str());
48 if (!infile.is_open())
49 {
50 std::cerr << "Error: file " << fname << " is not open in "
51 << "Objective::Objective(parametrized)" << std::endl;
52 exit(EXIT_FAILURE);
53 }
54 utils::find_word(infile, "oname");
55 infile >> oname;
56 utils::find_word(infile, "data");
57 std::string s;
58 getline(infile, s);
59
60 while (true)
61 {
62 double xin, yin, win;
63
64 if (x_is_present && w_is_present)
65 {
66 if ( !(infile >> xin >> yin >> win) ) break;
67 x.emplace_back(std::move(xin));
68 y.emplace_back(std::move(yin));
69 w.emplace_back(std::move(win));
70 continue;
71 }
72
73 if (!x_is_present && w_is_present)
74 {
75 if ( !(infile >> yin >> win) ) break;
76 y.emplace_back(std::move(yin));
77 w.emplace_back(std::move(win));
78 continue;
79 }
80
81 if (x_is_present && !w_is_present)
82 {
83 if ( !(infile >> xin >> yin) ) break;
84 x.emplace_back(std::move(xin));
85 y.emplace_back(std::move(yin));
86 w.emplace_back(1.0);
87 continue;
88 }
89
90 if (!x_is_present && !w_is_present)
91 {
92 if ( !(infile >> yin) ) break;
93 y.emplace_back(std::move(yin));
94 w.emplace_back(1.0);
95 continue;
96 }
97 }
98 infile.close();
99 infile.clear();

104double Objective::abs_diff(const Objective &o) const
105{
106 const size_t n = get_size();
107 const size_t na = o.get_size();
108 double d = fabs(static_cast<double>(get_size()) - static_cast<double>(na));
109 const size_t nmin = std::min(n, na);
110 for (size_t i = 0; i < nmin; ++i) d += fabs(y.at(i) - o.y.at(i));
111 return d;
112}
