36Goal::Goal(const bool analysis_in, const std::string &goal_path):
37 path(goal_path), obj(),
38 analysis(analysis_in), fitness(), best_fitness(-BIG), best_case(0), n(0)
39{
40 if (!analysis) return;
41 std::string fname(goal_path + "list.txt");
42 std::ifstream infile(fname.c_str());
43 if (!infile.is_open())
44 {
45 std::cerr << "Error: file " << fname << " is not open in "
46 << "Goal::Goal(parametrized)" << std::endl;
47 exit(EXIT_FAILURE);
48 }
49 std::string oname, xstr, wstr, sstr;
50 double weight;
51 utils::find_word(infile, "Objectives");
52 infile >> n;
53 utils::find_line(infile,
54 "Objective_Name weight x_is_present w_is_present rescale");
55 obj.reserve(n);
56 int i;
57 int nn = static_cast<int>(n);
58 for (i = 0; i < nn; ++i)
59 {
60 infile >> oname >> weight >> xstr >> wstr >> sstr;
61 obj.emplace_back(std::make_shared<Objective>(oname, path, i, weight,
62 utils::string_to_bool(xstr), utils::string_to_bool(wstr),
63 utils::string_to_bool(sstr)));
64 fitness.emplace_back(-BIG);
65 }
66 infile.close();
67 infile.clear();
68}
