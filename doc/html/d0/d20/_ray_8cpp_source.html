123 void Ray::trace(const Grid &g, const Mesh &m)
124 {
125  Progress counter("Ray_trace", level, 0, freq, "", std::cout);
126  counter.advance(); // final b.Sphere waypoint already pushed onto RayStack
127  RetIntercept intrcpt;
128  Waypoint w;
129  w.outface = wpt.top().outface;
130  while (true)
131  {
132  if (tracking) ++nzones;
133  const Zone *z = m.get_zone(zid);
134  intrcpt = z->hit(g, r, v, w.outface);
135  w.hitpt = intrcpt.w;
136  w.inface = intrcpt.fid;
137  w.outface = m.next_face(g, intrcpt);
138  r = w.hitpt;
139  if (w.inface == Face::BOUNDING_SPHERE) break;
140  zid = w.outface.my_zone;
141  wpt.push(w);
142  counter.advance();
143  }
144  nzd = utils::ndigits(nzones);
145  v.reverse();
146 }

150 void Ray::transport(const double ct) // transport this->y across *this
151 {
152  // Optically-thin formula used for optical depth < 2.0e-10
153  const double TR_THIN = exp(-2.0e-10);
154  #include "ArrDblLoop.inc"
155  {
156  double op = ab.at(i) + sc.at(i); // opacity
157  double tr = exp( - op * ct ); // transmission
158  double se; // self_emission
159  if (tr > TR_THIN)
160  se = em.at(i) * ct; // optically thin limit in 1-D
161  else
162  se = (1.0 - tr) * em.at(i) / op; // general 1-D solution
163  y.at(i) = y.at(i) * tr + se; // transmitted_backlighter + self_emission
164  }
165 }

169 void Ray::cross_Zone(const Zone *z, const Database &d, const Table &tbl,
170  const std::string &symmetry, const size_t ix,
171  const int ndmesh)
172 { // assumes that this->zid has already been set; e.g., in cross_Mesh()
173  const Waypoint &w = wpt.top();
174  const Vector3d r_old = r;
175  r = w.hitpt;
176  const double ct = (r - r_old).norm(); // chord length across current Zone
177  Zone *zz = const_cast<Zone *>(z);
178  if (d.get_tops_cmnd() == "none")
179  {
180  zz->load_spectra(d, tbl, ite, itr, ine, symmetry, ix, analysis,
181  jmin, jmax, em, ab, sc);
182  transport(ct);
183  }
184  wpt.pop();
185 
186  if (tracking)
187  {
188  distance += ct;
189  std::string izlabel("iz" + utils::int_to_string(iz, '0', nzd));
190  std::string izstring("iz " + utils::int_to_string(iz, ' ', nzd));
191  std::string header(hroot);
192  utils::replace_in_string(header, "\n", "_"+izlabel+"\n");
193  header += "\n" + izstring + utils::double_to_string(distance) + " cm";
194  header += "\nZoneID " + utils::int_to_string(z->get_id(), ' ', ndmesh);
195  header += "\ndata in W/cm2/sr/eV";
196  std::string fname(froot + "_" + izlabel + ".txt");
197  y.to_file(fname, header);
198  ++iz;
199  }
200 }
