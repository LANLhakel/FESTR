131 void Ray::trace(const Grid &g, const Mesh &m)
132 {
133  Progress counter("Ray_trace", level, 0, freq, "", std::cout);
134  counter.advance(); // final b.Sphere waypoint already pushed onto RayStack
135  RetIntercept intrcpt;
136  Waypoint w;
137  w.outface = wpt.top().outface;
138  while (true)
139  {
140  if (tracking) ++nzones;
141  auto z = m.get_zone(zid);
142  intrcpt = z->hit(g, r, v, w.outface);
143  w.hitpt = intrcpt.w;
144  w.inface = intrcpt.fid;
145  w.outface = m.next_face(g, intrcpt);
146  r = w.hitpt;
147  if (w.inface == Face::BOUNDING_SPHERE) break;
148  zid = w.outface.my_zone;
149  wpt.push(w);
150  counter.advance();
151  }
152  nzd = utils::ndigits(nzones);
153  v.reverse();
154 }

158 void Ray::transport(const double ct) // transport this->y across *this
159 {
160  // Optically-thin formula used for optical depth < 2.0e-10
161  const double TR_THIN = exp(-2.0e-10);
162 /*
163  #ifdef OMP
164  #pragma omp simd
165  #endif
166 */
167  for (size_t i = 0; i < n; ++i)
168  {
169  double op = ab[i] + sc[i]; // opacity
170  double tr = exp( - op * ct ); // transmission
171  double se; // self_emission
172  if (tr > TR_THIN)
173  se = em[i] * ct; // optically thin limit in 1-D
174  else
175  se = (1.0 - tr) * em[i] / op; // general 1-D solution
176  y[i] = y[i] * tr + se; // transmitted_backlighter + self_emission
177  }
178 }

182 void Ray::cross_Zone(ZonePtr z, const Database &d, const Table &tbl,
183  const std::string &symmetry, const size_t ix,
184  const int ndmesh)
185 { // assumes that this->zid has already been set; e.g., in cross_Mesh()
186  const Waypoint &w = wpt.top();
187  const Vector3d r_old = r;
188  r = w.hitpt;
189  const double ct = (r - r_old).norm(); // chord length across current Zone
190  if (d.get_tops_cmnd() == "none")
191  {
192  z->load_spectra(d, tbl, ite, itr, ine, symmetry, ix, analysis,
193  jmin, jmax, em, ab, sc);
194  transport(ct);
195  }
196  wpt.pop();
197 
198  if (tracking)
199  {
200  distance += ct;
201  std::string izlabel("iz" + utils::int_to_string(iz, '0', nzd));
202  std::string izstring("iz " + utils::int_to_string(iz, ' ', nzd));
203  std::string header(hroot);
204  utils::replace_in_string(header, "\n", "_"+izlabel+"\n");
205  header += "\n" + izstring + utils::double_to_string(distance) + " cm";
206  header += "\nZoneID " + utils::int_to_string(z->get_id(), ' ', ndmesh);
207  header += "\ndata in W/cm2/sr/eV";
208  std::string fname(froot + "_" + izlabel + ".txt");
209  y.to_file(fname, header);
210  ++iz;
211  }
212 }
