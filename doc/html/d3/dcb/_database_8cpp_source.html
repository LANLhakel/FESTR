23 void Database::read_grid_file(const std::string &fname, const int iwidth,
24  size_t &nbits, size_t &n,
25  std::vector<double> &v,
26  std::vector<std::string> &v_str)
27 {
28  std::ifstream infile(fname.c_str());

30  {
31  std::cerr << "Error: file " << fname << " is not open in "

33  exit(EXIT_FAILURE);
34  }
35  std::ifstream infile_str(fname.c_str());
36  if (!infile_str.is_open())
37  {
38  std::cerr << "Error: file " << fname << " (str) is not open in "
39  << "Database::read_grid_file" << std::endl;
40  exit(EXIT_FAILURE);
41  }
42  utils::find_line(infile, "Number of bits:");
43  infile >> nbits;
44  utils::find_line(infile, "Number of grid points:");
45  infile >> n;
46  utils::find_line(infile, "Grid points:");
47  utils::find_line(infile_str, "Grid points:");
48  for (size_t i = 0; i < n; ++i)
49  {
50  size_t j, k;
51  double x;
52  std::string s;
53  infile >> j;
54  infile_str >> j;
55  if (iwidth == 0) // ne_man
56  {
57  infile >> x;

59  infile_str >> s;
60  v_str.emplace_back(utils::trim(s));
61  }
62  else // iwidth > 0
63  {
64  infile >> k;
65  v.emplace_back(std::move(static_cast<double>(k)));
66  v_str.emplace_back(utils::int_to_string(k, '0', iwidth));
67  }
68  }

70  infile.clear();

72  infile_str.clear();

109  nbits_ne(0), nne(0), ne(), ne_str(),

111 {
112  if (t == "none")
113  {
114  read_grid_file(p + "grids/te_grid.txt", 0, nbits_te, nte, te, te_str);
115  read_grid_file(p + "grids/tr_grid.txt", 0, nbits_tr, ntr, tr, tr_str);
116  read_grid_file(p + "grids/ne_man_grid.txt", 0,
117  nbits_neman, nneman, neman, neman_str);
118  read_grid_file(p + "grids/ne_exp_grid.txt", 2,
119  nbits_neexp, nneexp, neexp, neexp_str);
120  // Assemble electron number density grid
121  nbits_ne = nbits_neman + nbits_neexp;
122  nne = nneman * nneexp;
123  for (size_t iexp = 0; iexp < nneexp; ++iexp)
124  {
125  double e = pow(10.0, neexp.at(iexp));
126  std::string estr = neexp_str.at(iexp);
127  for (size_t iman = 0; iman < nneman; ++iman)
128  {
129  double x = neman.at(iman) * e;
130  ne.emplace_back(std::move(x));
131  std::string s = neman_str.at(iman) + "e" + estr;
132  ne_str.emplace_back(std::move(s));
133  }
134  }
135  } // t == "none"

137  size_t nbits_hv;
138  std::vector<std::string> hv_str;
139  read_grid_file(p + "grids/hv_grid.txt", 0, nbits_hv, nhv, hv, hv_str);
