72 {

136 { // pmh_2014_1125

138  int first_turn = NOT_INITIALIZED;
139  int turn;
140 
141  // build local coordinate system in the plane of *this
142  const Vector3d Zhat = normal(g);
143  const Vector3d origin = POINT(0);
144  const Vector3d Xhat = (POINT(1) - origin).normalize();
145  const Vector3d Yhat = Zhat % Xhat;
146 
147  // build q as the representation of v in this local coordinate system
148  const Vector3d p = w - origin;
149  const Vector3d q(p*Xhat, p*Yhat, 0.0);
150  Vector3d head3d, tail3d, head, tail;
151 
152  // loop over *this Polygon's sides
153  size_t i;
154  const size_t n = size();
155  for (size_t j = 0; j < n; ++j)
156  { // see if q is on the same side relative to each side's vector
157  i = j + 1;
158  if (i == n) i = 0;
159  head3d = POINT(i) - origin;
160  tail3d = POINT(j) - origin;
161  head = Vector3d(head3d*Xhat, head3d*Yhat, 0.0);
162  tail = Vector3d(tail3d*Xhat, tail3d*Yhat, 0.0);
163  turn = q.get_turn(tail, head);
164 
165  if (turn != 0)
166  { // points lying exactly on the line are not considered
167  if (first_turn == NOT_INITIALIZED)
168  {
169  first_turn = turn;
170  }
171  else
172  { // if turn differs from first_turn => v is outside of *this
173  if (first_turn != turn) return false;
