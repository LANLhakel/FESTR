208 void Hydro::load_at(const size_t i, Grid &g, Mesh &m) const
209 {
210  std::string tlabel;
211  if (analysis) // analysis of case #i
212  {
213  tlabel = "0";
214  if (g.size() == 0) g.load(path, tlabel);
215  if (m.size() == 0) m.load(path, tlabel);
216 
217  if (symmetry == "spherical")
218  {
219  std::pair<size_t, size_t> pr = utils::one_to_two(ndim, i);
220  size_t izone = pr.first;
221  size_t icase = pr.second;
222  auto cptr = cell.at(izone);
223  CellEOS e = cptr->get_cell_eos(icase);
224  auto zptr = m.get_zone(izone);
225  zptr->set_te(e.te);
226  zptr->set_tr(e.tr);
227  zptr->set_nmat(e.nmat);
228  zptr->set_mat(e.mat);
229  zptr->set_fp(e.fp);
230  }
231  else // symmetry == "none"
232  {
233  std::vector<size_t> indx = utils::one_to_many(ndim, i);
234  // seed Mesh with CellEOS data
235  for (size_t izone = 0; izone < nzones; ++izone)
236  {
237  auto cptr = cell.at(izone);
238  CellEOS e = cptr->get_cell_eos(indx.at(izone));
239  auto zptr = m.get_zone(izone);
240  zptr->set_te(e.te);
241  zptr->set_tr(e.tr);
242  zptr->set_nmat(e.nmat);
243  zptr->set_mat(e.mat);
244  zptr->set_fp(e.fp);
245  }
246  }
247  }
248  else // postprocessing time interval [i, i+1]
249  {
250  tlabel = utils::int_to_string(i, '0', ntd);
251  g.load(path, tlabel);
252  m.load(path, tlabel);
253  }
254 }
