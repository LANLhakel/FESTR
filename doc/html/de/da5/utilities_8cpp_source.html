109 utils::RetQuad utils::solve_quadratic(const double a, const double b,
110  const double c, const double EQT)
111 { // Numerical Recipes in FORTRAN, Second Edition
112  const double BIG = -1.0e100;
113  const double TOL = fabs(EQT);
114  RetQuad rv;
115 
116  if (fabs(a) <= TOL) // a == 0 (linear equation)
117  {
118  if (fabs(b) <= TOL) // b == 0 (return no solution)
119  {
120  rv.nroots = 0;
121  rv.x1 = rv.x2 = BIG;
122  }
123  else // b != 0
124  {
125  rv.nroots = 1;
126  rv.x1 = rv.x2 = -c/b;
127  }
128  }
129  else // a != 0 (quadratic equation)
130  {
131  double d = b*b - 4.0*a*c;
132 
133  if (d < 0.0) // no real solution
134  {
135  rv.nroots = 0;
136  rv.x1 = rv.x2 = BIG;
137  }
138  else // d >= 0.0
139  {
140  if (fabs(b) <= TOL) // this extra b == 0 branch is needed
141  { // because Numerical Recipes assumes that sgn(b) cannot be 0
142  // whereas our sign_eqt(b, TOL) == 0 if b == 0 within TOL
143  rv.x1 = sqrt(-c/a);
144  rv.x2 = -rv.x1;
145  }
146  else
147  { // sign_eqt(b, TOL) != 0 in this branch
148  double q = -0.5 * (b + sign_eqt(b, TOL) * sqrt(d));
149  rv.x1 = q/a;
150  rv.x2 = c/q;
151  }
152 
153  if (rv.x1 < rv.x2) std::swap(rv.x1, rv.x2); // ensure x1 >= x2
154 
155  if (d <= TOL)
156  rv.nroots = 1;
157  else
158  rv.nroots = 2;
159 
160  } // d < 0.0
161 
162  } // fabs(a) <= TOL
163 
164  return rv;
165 }

416 {

418  std::vector<size_t> indx;
419  const size_t ONE(1);
420  size_t n =
421  std::accumulate(dim.begin(), dim.end(), ONE, std::multiplies<size_t>() );
422  if (j >= n) throw std::out_of_range("out of range exception");
423 
424  size_t remainder;

426  size_t i = ndim;

429  {
430  --i;

432  indx.at(i) = remainder;
433  j = (j - remainder) / dim[i];
