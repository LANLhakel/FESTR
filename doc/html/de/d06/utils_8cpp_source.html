131 utils::RetQuad utils::solve_quadratic(const double a, const double b,
132  const double c, const double EQT)
133 { // Numerical Recipes in FORTRAN, Second Edition
134  const double BIG = -1.0e100;
135  const double TOL = fabs(EQT);
136  RetQuad rv;
137 
138  if (fabs(a) <= TOL) // a == 0 (linear equation)
139  {
140  if (fabs(b) <= TOL) // b == 0 (return no solution)
141  {
142  rv.nroots = 0;
143  rv.x1 = rv.x2 = BIG;
144  }
145  else // b != 0
146  {
147  rv.nroots = 1;
148  rv.x1 = rv.x2 = -c/b;
149  }
150  }
151  else // a != 0 (quadratic equation)
152  {
153  double d = b*b - 4.0*a*c;
154  if (fabs(d) <= TOL) d = 0.0;
155 
156  if (d < 0.0) // no real solution
157  {
158  rv.nroots = 0;
159  rv.x1 = rv.x2 = BIG;
160  }
161  else // d >= 0.0
162  {
163  if (fabs(b) <= TOL) // this extra b == 0 branch is needed
164  { // because Numerical Recipes assumes that sgn(b) cannot be 0
165  // whereas our sign_eqt(b, TOL) == 0 if b == 0 within TOL
166  rv.x1 = sqrt(-c/a);
167  rv.x2 = -rv.x1;
168  }
169  else
170  { // sign_eqt(b, TOL) != 0 in this branch
171  double q = -0.5 * (b + sign_eqt(b, TOL) * sqrt(d));
172  rv.x1 = q/a;
173  rv.x2 = c/q;
174  }
175 
176  if (rv.x1 < rv.x2) std::swap(rv.x1, rv.x2); // ensure x1 >= x2
177 
178  if (d <= TOL)
179  rv.nroots = 1;
180  else
181  rv.nroots = 2;
182 
183  } // d < 0.0
184 
185  } // fabs(a) <= TOL
186 
187  return rv;
188 }
