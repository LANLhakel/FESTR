746 template<typename Tx, typename Ty>
747 // T's must support "double operator [](size_t)" method
748 // x grid is assumed to be monotonically increasing
749 double integrate_trapezoid(double a, double b,
750  const Tx &x, const Ty &y, const size_t n)
751 {
752  if (sign_eqt(a-b, 1.0e-20) == 0) return 0.0;
753  // if (a == b) return 0.0;
754 
755  double sign;
756  if (a > b)
757  {
758  sign = -1.0;
759  std::swap(a, b);
760  }
761  else
762  sign = 1.0;
763 
764  size_t nm1 = n - 1;
765  a = std::max(a, x[0]);
766  b = std::min(b, x[nm1]);
767 
768  size_t ia = bisect(a, x, 0, nm1);
769  double ya = fitlin(a, x[ia], y[ia], x[ia+1], y[ia+1]);
770 
771  size_t ib = bisect(b, x, ia, nm1);
772  double yb = fitlin(b, x[ib], y[ib], x[ib+1], y[ib+1]);
773 
774  if (ia == ib) return sign * (b - a) * (yb + ya) / 2.0;
775 
776  double s = (x[ia+1] - a) * (y[ia+1] + ya) + (b - x[ib]) * (yb + y[ib]);
777  for (size_t i = ia + 1; i < ib; ++i)
778  s += (x[i+1] - x[i]) * (y[i+1] + y[i]);
779 
780  return s * sign / 2.0;
781 }

798 // T's must support "double operator [](size_t)", "size_t size(void)" methods
799 void convolution(const double fwhm, const Txin &xin, const Tyin &yin,
800  const Txout &xout, Tyout &yout)
801 {
802  const double FWHM_TO_SIGMA = 0.5 / sqrt(2.0 * log(2.0));
803 
804  size_t nxin = xin.size();
805  size_t nyin = yin.size();
806  size_t nxout = xout.size();
807  size_t nyout = yout.size();
808 
809  if ( (nxin != nyin) || (nxout != nyout))
810  {
811  std::string message = "Error: convolution ranges do not match:";
812  message += "\nsize(xin):" + int_to_string(nxin, ' ', cnst::INT_WIDTH);
813  message += "\nsize(yin):" + int_to_string(nyin, ' ', cnst::INT_WIDTH);
814  message += "\nsize(xout):" + int_to_string(nxout, ' ', cnst::INT_WIDTH);
815  message += "\nsize(yout):" + int_to_string(nyout, ' ', cnst::INT_WIDTH);
816  throw std::range_error(message);
817  }
818 
819  std::string xmode("lin");
820  std::string ymode("log"); // switch to "lin", if any(yin) <= 0
821  double dx;
822  double dxmin = 1.0e99; // find the smallest grid increment in xin
823  for (size_t i = 0; i < nxin-1; ++i)
824  {
825  dx = fabs(xin.at(i+1) - xin.at(i));
826  if (dx < dxmin) dxmin = dx;
827  if (yin.at(i) <= 0.0) ymode = "lin";
828  }
829  if (yin.at(nxin-1) <= 0.0) ymode = "lin";
830  if (fwhm <= 1.5*dxmin) // assign yout = yin
831  {
832  syngrids(xin, yin, nxin, xmode, ymode, xout, yout, nxout);
833  return;
834  }
835 
836  double sigma = fwhm * FWHM_TO_SIGMA;
837 
838  // extrapolate xin, yin at the edges to preempt artificial drop
839  size_t ninc = 100; // must be even
840  size_t nhalf = ninc / 2; // add nhalf points to the grid on each side
841  size_t nxext = nxin + ninc;
842  std::vector<double> xext(nxext), yext(nxext); // extrapolated xin, yin
843  double ds = sigma / 10.0; // auxiliary grid spacing
844  for (size_t j = 0; j < nxext; ++j)
845  {
846  if (j < nhalf) // left of the original grid xin
847  {
848  dx = ds * static_cast<double>(nhalf - j);
849  xext.at(j) = xin.at(0) - dx;
850  yext.at(j) = fitpoint(xext.at(j),
851  xin.at(0), yin.at(0),
852  xin.at(1), yin.at(1),
853  xmode, ymode);
854  }
855  else if (j >= nxext - nhalf) // right of the original grid xin
856  {
857  xext.at(j) = xext.at(j-1) + ds;
858  yext.at(j) = fitpoint(xext.at(j),
859  xin.at(nxin-2), yin.at(nxin-2),
860  xin.at(nxin-1), yin.at(nxin-1),
861  xmode, ymode);
862  }
863  else // on the original grid xin
864  {
865  xext.at(j) = xin.at(j - nhalf);
866  yext.at(j) = yin.at(j - nhalf);
867  }
868  }
869 
870  std::vector<double> y(nxext); // convolution integrand
871  for (size_t i = 0; i < nxout; ++i) // loop over output grid points
872  {
873  for (size_t j = 0; j < nxext; ++j) // build convolution integrand
874  y.at(j) = yext.at(j) * gaussian(xout.at(i), xext.at(j), sigma);
875
