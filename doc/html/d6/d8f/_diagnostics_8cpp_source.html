294 void Diagnostics::analyze(const Database &d, Hydro &h, Goal &gol)
295 {
296  #include <diag_exec.inc>
297  std::vector<size_t> ndim = h.get_ndim();
298  size_t n = ndim.size();
299  size_t current_zone;
300  size_t previous_zone = n;
301  std::string name("Case");
302 
303  { // begin block defining Progress diag_counter scope
304  Progress diag_counter(name, level, nintervals, freq, DIAGSEP, std::cout);
305 
306  for (size_t j = 0; j < nintervals; ++j) // loop over cases
307  {
308  size_t it = h.time_index_at(j);
309  double t = h.time_at(0);
310  double dt = h.dt_at(0);
311 
312  if (det.at(0).get_symmetry() == "spherical")
313  {
314  size_t jt = nintervals - it - 1; // peeling the onion
315  h.load_at(jt, g, m); // using summation-based index jt
316  if (it == 0)
317  { // set bounding Zone (fixed) on first pass through it loop
318  previous_zone = n - 1;
319  h.load_at(0, g, m);
320  }
321  else // it > 0
322  {
323  current_zone = utils::one_to_two(ndim, jt).first;
324  if (current_zone != previous_zone)
325  {
326  auto obj = gol.get_objective(previous_zone-1);
327  h.load_at(obj->get_best_case(), g, m);
328  previous_zone = current_zone;
329  }
330  }
331  det.at(0).evaluate(ndim, g, m, d, tbl, jt, t, dt, ntd, gol);
332  }
333 
334  else // "symmetry == none" path
335  {
336  h.load_at(it, g, m); // using product-based index it
337  Progress det_counter("Detector", level+1, ndet,
338  diag_counter.get_next_freq(freq_det),
339  DETSEP, std::cout);
340  for (size_t id = 0; id < ndet; ++id) // loop over Detectors
341  {
342  det.at(id).do_patches(&det_counter, g, m, d, tbl, it, t, dt,
343  ntd, gol);
344  det_counter.advance();
345  }
346  } // det.at(0).get_symmetry()
347 
348  gol.update_best(it);
349  diag_counter.advance();
350  } // end loop over cases
351  } // end block defining Progress diag_counter scope
352 
353  // redo the best case and write results
354  gol.set_analysis(false);
355 
356  // switching from summation-based indices jt to product-based index it,
357  // as if we hadn't taken advantage of spherical symmetry during search
358  if (h.get_symmetry() == "spherical") h.set_symmetry("none");
359 
360  size_t it = gol.get_best_case();
361  h.load_at(it, g, m);
362  double t = h.time_at(0);
363  double dt = h.dt_at(0);
364  ntd = h.get_ntd();
365  for (size_t id = 0; id < ndet; ++id) // loop over Detectors
366  det.at(id).do_patches(nullptr, g, m, d, tbl, it, t, dt, ntd, gol);
367 
368  gol.set_analysis(true);
369  size_t nzones = m.size();
370  std::string zone_content;
371  std::string best_case_fname(outpath + "best_case.dat");
372  std::ofstream best_case_plot(best_case_fname.c_str());
373  best_case_fname = outpath + "best_case.txt";
374  std::ofstream best_case_full(best_case_fname.c_str());
375  best_case_full << "time 0 s";

379  std::cout << header;
