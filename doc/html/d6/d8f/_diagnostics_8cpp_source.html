537  std::string name("Case");
538 
539  { // begin block defining Progress diag_counter scope
540  Progress diag_counter(name, level, nt, freq, DIAGSEP, std::cout);
541  for (size_t it = 0; it < nt; ++it) // loop over cases
542  {
543  double t = h.time_at(0);
544  double dt = h.dt_at(0);
545 
546  if (det.at(0).get_symmetry() == "spherical")
547  {
548  size_t jt = nt - it - 1; // peeling the onion
549  h.load_at(jt, g, m); // using summation-based index jt
550  if (it == 0)
551  { // set bounding Zone (fixed) on first pass through it loop
552  previous_zone = n - 1;
553  h.load_at(0, g, m);
554  }
555  else // it > 0
556  {
557  current_zone = utils::one_to_two(ndim, jt).first;
558  if (current_zone != previous_zone)
559  {
560  const Objective *obj = gol.get_objective(previous_zone-1);
561  h.load_at(obj->get_best_case(), g, m);
562  previous_zone = current_zone;
563  }
564  }
565  det.at(0).evaluate(ndim, g, m, d, tbl, jt, t, dt, ntd, gol);
566  }
567 
568  else // "symmetry == none" path
569  {
570  h.load_at(it, g, m); // using product-based index it
571  Progress det_counter("Detector", level+1, ndet,
572  diag_counter.get_next_freq(freq_det),
573  DETSEP, std::cout);

575  {

577  ntd, gol);
578  det_counter.advance();
579  }
580  } // det.at(0).get_symmetry()
581

584  } // end loop over cases

586

589
