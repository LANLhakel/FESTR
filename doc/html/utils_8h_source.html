221template<typename T> // T must support "double operator [](size_t)" method
222static size_t bisect(const double x, const T &v, size_t ilow, size_t ihigh)
223{
224 size_t imid;
225 while (ihigh - ilow > 1)
226 {
227 imid = (ihigh + ilow) / 2;
228 if (v[imid] <= x)
229 ilow = imid;
230 else
231 ihigh = imid;
232 }
233 return ilow;
234}

246template<typename T> // T must support "double operator [](size_t)" method
247static void bracket(const double x, const T &v,
248 const size_t imax, size_t &ilow, size_t &ihigh)
249{
250 size_t m = 1;
251 if (x < v[ilow])
252 {
253 while (true)
254 {
255 ihigh = ilow;
256 if (ilow > m)
257 {
258 ilow -= m;
259 if (v[ilow] <= x) return;
260 }
261 else
262 {
263 ilow = 0;
264 return;
265 }
266 m *= 2;
267 }
268 }
269 else if (v[ihigh] <= x)
270 {
271 while (true)
272 {
273 ilow = ihigh;
274 ihigh += m;
275 if (ihigh > imax)
276 {
277 ihigh = imax;
278 return;
279 }
280 if (x < v[ihigh]) return;
281 m *= 2;
282 }
283 }
284}

297template<typename T> // T must support "double operator [](size_t)" method
298static size_t nearest(const double x, const T &v,
299 const size_t n, const size_t i)
300{
301 if (n <= 1) return 0;
302 if (x <= v[0]) return 0;
303 size_t imax = n - 1;
304 if (x >= v[imax]) return imax;
305
306 size_t ilow, ihigh;
307 if (i == 0)
308 ilow = 0;
309 else
310 ilow = i - 1;
311 if (i == imax)
312 ihigh = imax;
313 else
314 ihigh = i + 1;
315
316 bracket(x, v, imax, ilow, ihigh);
317 size_t j = bisect(x, v, ilow, ihigh);
318 size_t jp1 = j + 1; // 0 <= j < imax is guaranteed by bisect()
319 if (v[jp1] - x < x - v[j])
320 return jp1;
321 else
322 return j;
323}

495template<typename Txin, typename Tyin, typename Txout, typename Tyout>
496// T's must support "double operator [](size_t)" methods
497static void syngrids(const Txin &xin, const Tyin &yin, const size_t NIN,
498 const std::string &xmode, const std::string &ymode,
499 const Txout &xout, Tyout &yout, const size_t NOUT)
500{
501 if (NOUT == 0) return;
502 if (NIN == 0)
503 {
504 for (size_t i = 0; i < NOUT; ++i) yout[i] = 0.0;
505 return;
506 }
507 const size_t MAXIN = NIN - 1;
508 const double XMIN = xin[0];
509 const double YMIN = yin[0];
510 const double XMAX = xin[MAXIN];
511 const double YMAX = yin[MAXIN];
512 size_t j;
513 size_t ilow = 0;
514 size_t ihigh = 1;
515
516 for (size_t i = 0; i < NOUT; ++i)
517 {
518 double x = xout[i];
519
520 if (x <= XMIN)
521 {
522 yout[i] = YMIN; // flat extrapolation
523 continue;
524 }
525
526 if (x >= XMAX)
527 {
528 yout[i] = YMAX; // flat extrapolation
529 continue;
530 }
531
532 bracket(x, xin, MAXIN, ilow, ihigh);
533 j = bisect(x, xin, ilow, ihigh);
534 yout[i] = fitpoint(x, xin[j], yin[j], xin[j+1], yin[j+1],
535 xmode, ymode);
536 }

738template<typename Tx, typename Ty>
739// T's must support "double operator [](size_t)" method
740// x grid is assumed to be monotonically increasing
741static double integrate_trapezoid(double a, double b,
742 const Tx &x, const Ty &y, const size_t n)
743{
744 if (sign_eqt(a-b, 1.0e-20) == 0) return 0.0;
745 // if (a == b) return 0.0;
746
747 double sign;
748 if (a > b)
749 {
750 sign = -1.0;
751 std::swap(a, b);
752 }
753 else
754 sign = 1.0;
755
756 size_t nm1 = n - 1;
757 a = std::max(a, x[0]);
758 b = std::min(b, x[nm1]);
759
760 size_t ia = bisect(a, x, 0, nm1);
761 double ya = fitlin(a, x[ia], y[ia], x[ia+1], y[ia+1]);
762
763 size_t ib = bisect(b, x, ia, nm1);
764 double yb = fitlin(b, x[ib], y[ib], x[ib+1], y[ib+1]);
765
766 if (ia == ib) return sign * (b - a) * (yb + ya) / 2.0;
767
768 double s = (x[ia+1] - a) * (y[ia+1] + ya) + (b - x[ib]) * (yb + y[ib]);
769 for (size_t i = ia + 1; i < ib; ++i)
770 s += (x[i+1] - x[i]) * (y[i+1] + y[i]);
771
772 return s * sign / 2.0;
773}

789template<typename Txin, typename Tyin, typename Txout, typename Tyout>
790// T's must support "double operator [](size_t)", "size_t size()" methods
791static void convolution(const double fwhm, const Txin &xin, const Tyin &yin,
792 const Txout &xout, Tyout &yout)
793{
794 const double FWHM_TO_SIGMA = 0.5 / sqrt(2.0 * log(2.0));
795
796 size_t nxin = xin.size();
797 size_t nyin = yin.size();
798 size_t nxout = xout.size();
799 size_t nyout = yout.size();
800
801 if ( (nxin != nyin) || (nxout != nyout))
802 {
803 std::string message = "Error: convolution ranges do not match:";
804 message += "\nsize(xin):" + int_to_string(nxin, ' ', cnst::INT_WIDTH);
805 message += "\nsize(yin):" + int_to_string(nyin, ' ', cnst::INT_WIDTH);
806 message += "\nsize(xout):" + int_to_string(nxout, ' ', cnst::INT_WIDTH);
807 message += "\nsize(yout):" + int_to_string(nyout, ' ', cnst::INT_WIDTH);
808 throw std::range_error(message);
809 }
810
811 std::string xmode("lin");
812 std::string ymode("log"); // switch to "lin", if any(yin) <= 0
813 double dx;
814 double dxmin = 1.0e99; // find the smallest grid increment in xin
815 for (size_t i = 0; i < nxin-1; ++i)
816 {
817 dx = fabs(xin.at(i+1) - xin.at(i));
818 if (dx < dxmin) dxmin = dx;
819 if (yin.at(i) <= 0.0) ymode = "lin";
820 }
821 if (yin.at(nxin-1) <= 0.0) ymode = "lin";
822 if (fwhm <= 1.5*dxmin) // assign yout = yin
823 {
824 syngrids(xin, yin, nxin, xmode, ymode, xout, yout, nxout);
825 return;
826 }
827
828 double sigma = fwhm * FWHM_TO_SIGMA;
829
830 // extrapolate xin, yin at the edges to preempt artificial drop
831 size_t ninc = 100; // must be even
832 size_t nhalf = ninc / 2; // add nhalf points to the grid on each side
833 size_t nxext = nxin + ninc;
834 std::vector<double> xext(nxext), yext(nxext); // extrapolated xin, yin
835 double ds = sigma / 10.0; // auxiliary grid spacing
836 for (size_t j = 0; j < nxext; ++j)
837 {
838 if (j < nhalf) // left of the original grid xin
839 {
840 dx = ds * static_cast<double>(nhalf - j);
841 xext.at(j) = xin.at(0) - dx;
842 yext.at(j) = fitpoint(xext.at(j),
843 xin.at(0), yin.at(0),
844 xin.at(1), yin.at(1),
845 xmode, ymode);
846 }
847 else if (j >= nxext - nhalf) // right of the original grid xin
848 {
849 xext.at(j) = xext.at(j-1) + ds;
850 yext.at(j) = fitpoint(xext.at(j),
851 xin.at(nxin-2), yin.at(nxin-2),
852 xin.at(nxin-1), yin.at(nxin-1),
853 xmode, ymode);
854 }
855 else // on the original grid xin
856 {
857 xext.at(j) = xin.at(j - nhalf);
858 yext.at(j) = yin.at(j - nhalf);
859 }
860 }
861
862 std::vector<double> y(nxext); // convolution integrand
863 for (size_t i = 0; i < nxout; ++i) // loop over output grid points
864 {
865 for (size_t j = 0; j < nxext; ++j) // build convolution integrand
866 y.at(j) = yext.at(j) * gaussian(xout.at(i), xext.at(j), sigma);
867
868 yout[i] = integrate_trapezoid(xext[0], xext[nxext-1], xext, y, nxext);
869 }
870}
