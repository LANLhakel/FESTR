179RetIntercept Surface::intercept(const Grid &g, const Vector3d &p,
180 const Vector3d &u, const double eqt,
181 const FaceID &fid) const
182{
183 RetIntercept rv, pt;
184 for (size_t i = 0; i < nfaces; ++i)
185 {
186 pt = face.at(i)->intercept(g, p, u, eqt, fid);
187 if (pt.is_found && pt.t < rv.t)
188 {
189 rv.t = pt.t;
190 rv.w = pt.w;
191 rv.fid = FaceID(i, -1);
192 rv.is_found = true;
193 }
194 }
195 return rv;
196}
